114  inline void setFisheye(const T fisheye) noexcept
115  {
116  T factor = 1. - Math<T>::clip(fisheye, (T)0., (T)1.);
117  for(ulong i = 0; i < Processor<Hoa2d, T>::Planewaves::getNumberOfPlanewaves(); i++)
118  {
119  T azimuth = (T)i / (T)Processor<Hoa2d, T>::Planewaves::getNumberOfPlanewaves() * HOA_2PI - HOA_PI;
120  if(azimuth < HOA_PI)
121  {
122  azimuth *= factor;
123  }
124  else
125  {
126  azimuth = HOA_2PI - ((HOA_2PI - azimuth) * factor);
127  }
128  m_encoders[i]->setAzimuth(azimuth);
129  }
130  }
