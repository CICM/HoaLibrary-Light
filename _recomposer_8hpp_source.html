92  inline void setFisheye(const T fisheye) noexcept
93  {
94  T factor = 1. - Math<T>::clip(fisheye, (T)0., (T)1.);
95  for(ulong i = 0; i < Processor<Hoa2d, T>::Planewaves::getNumberOfPlanewaves(); i++)
96  {
97  T azimuth = (T)i / (T)Processor<Hoa2d, T>::Planewaves::getNumberOfPlanewaves() * HOA_2PI - HOA_PI;
98  if(azimuth < HOA_PI)
99  {
100  azimuth *= factor;
101  }
102  else
103  {
104  azimuth = HOA_2PI - ((HOA_2PI - azimuth) * factor);
105  }
106  m_encoders[i]->setAzimuth(azimuth);
107  }
108  }
109

113  for(ulong i = 1; i < Processor<Hoa2d, T>::Planewaves::getNumberOfPlanewaves(); i++)
114  {
115  m_encoders[i]->processAdd(inputs++, outputs);
116  }
