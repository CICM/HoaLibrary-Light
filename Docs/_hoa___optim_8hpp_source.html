67  void setMode(Mode mode) hoa_noexcept
68  {
69  const size_t size = ProcessorHarmonics<D, T>::getNumberOfHarmonics();
70  const size_t order = ProcessorHarmonics<D, T>::getDecompositionOrder();
71  if(mode == Basic)
72  {
73  for(size_t i = 0; i < size; ++i) {
74  m_weights[i] = T(1.);
75  }
76  }
77  else if(mode == MaxRe)
78  {
79  m_weights[0] = T(1.);
80  for(size_t i = 1; i < size; i++) {
81  const size_t degree = ProcessorHarmonics<D, T>::getHarmonicDegree(i);
82  m_weights[i] = std::cos(T(degree) * T(HOA_PI) / T(2. * order + 2.));
83  }
84  }
85  else
86  {
87  m_weights[0] = T(1.);
88  const long double facn = ofactorial(long(order));
89  for(size_t i = 1; i < size; i++) {
90  const size_t degree = ProcessorHarmonics<D, T>::getHarmonicDegree(i);
91  m_weights[i] = T(facn / ofactorial(long(order - degree))) * T(facn / ofactorial(long(order + degree)));
92  }
93 
94  }

104  void process(T const* inputs, T* outputs) hoa_noexcept hoa_final
105  {
106  const size_t size = ProcessorHarmonics<D, T>::getNumberOfHarmonics();
107  const T* weights = m_weights;
108  for(size_t i = size>>3; i; --i, inputs += 8, weights += 8, outputs += 8)
109  {
110  T const f0 = inputs[0], f1 = inputs[1], f2 = inputs[2], f3 = inputs[3];
111  T const f4 = inputs[4], f5 = inputs[5], f6 = inputs[6], f7 = inputs[7];
112 
113  T const g0 = weights[0], g1 = weights[1], g2 = weights[2], g3 = weights[3];
114  T const g4 = weights[4], g5 = weights[5], g6 = weights[6], g7 = weights[7];
115 
116  outputs[0] = f0 * g0; outputs[1] = f1 * g1; outputs[2] = f2 * g2; outputs[3] = f3 * g3;
117  outputs[4] = f4 * g4; outputs[5] = f5 * g5; outputs[6] = f6 * g6; outputs[7] = f7 * g7;
118  }
119  for(size_t i = size&7; i; --i, inputs++, weights++, outputs++)
120  {

122  }
