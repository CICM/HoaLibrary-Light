219  template <typename T> class RecomposerFixe<Hoa2d, T> : public Recomposer<Hoa2d, T>
220  {
221  enum Recomposition
222  {
223  Fixe = 0,
224  Fisheye = 1,
225  Free = 2
226  };
227 
228  private:
229  T* m_matrix;
230 
231  public:
233 
237  RecomposerFixe(size_t order, size_t numberOfPlanewaves) hoa_noexcept : Recomposer<Hoa2d, T>(order, numberOfPlanewaves)
238  {
239  const T factor = 1.;
240  T* vector = Signal<T>::alloc(Encoder<Hoa2d, T>::getNumberOfHarmonics());
241  m_matrix = Signal<T>::alloc(ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves() * Encoder<Hoa2d, T>::getNumberOfHarmonics());
242  for(size_t i = 0; i < ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves(); i++)
243  {
244  EncoderBasic<Hoa2d, T>::setAzimuth(ProcessorPlanewaves<Hoa2d, T>::getPlanewaveAzimuth(i));
245  EncoderBasic<Hoa2d, T>::process(&factor, vector);
246  for(size_t j = 0; j < Encoder<Hoa2d, T>::getNumberOfHarmonics(); j++)
247  {
248  m_matrix[j * ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves() + i] = vector[j];
249  }
250  }
251  Signal<T>::free(vector);
252  }
253 
255 
257  ~RecomposerFixe()
258  {
259  Signal<T>::free(m_matrix);
260  }
261 
263 
267  void process(const T* inputs, T* outputs) hoa_noexcept hoa_override
268  {
269  Signal<T>::mul(ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves(), Encoder<Hoa2d, T>::getNumberOfHarmonics(), inputs, m_matrix, outputs);
270  }
271  };

308  inline void setFisheye(const T fisheye) hoa_noexcept hoa_override
309  {
310  T factor = 1. - Math<T>::clip(fisheye, (T)0., (T)1.);
311  for(size_t i = 0; i < ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves(); i++)
312  {
313  T azimuth = (T)i / (T)ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves() * HOA_2PI;
314  if(azimuth < HOA_PI)
315  {
316  azimuth *= factor;
317  }
318  else
319  {
320  azimuth = HOA_2PI - ((HOA_2PI - azimuth) * factor);
321  }
322  m_encoders[i]->setAzimuth(azimuth);
323  }
324  }
325 
327 
331  inline void process(const T* inputs, T* outputs) hoa_noexcept hoa_override
332  {
333  m_encoders[0]->process(inputs, outputs);
334  for(size_t i = 1; i < ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves(); i++)
335  {
336  m_encoders[i]->processAdd(++inputs, outputs);
337  }
338  }
339  };
