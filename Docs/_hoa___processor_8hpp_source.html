134  generateCicle(nplws);

307  void generateTetrahedron() hoa_noexcept
308  {
309  const double oh = -(sqrt(2. / 3.) / sqrt(3. / 8.) - 1.);
310  const double hc = sqrt(1. - oh * oh);
311  const double el = asin(oh / sqrt(hc*hc + oh*oh));
312  m_planewaves.push_back(Planewave<D, T>(1ul, 0., HOA_PI2));
313  m_planewaves.push_back(Planewave<D, T>(2ul, 0., el));
314  m_planewaves.push_back(Planewave<D, T>(3ul, T(HOA_2PI / 3.), el));
315  m_planewaves.push_back(Planewave<D, T>(4ul, T(HOA_2PI / 1.5), el));
316  }

331  void generateOctahedron() hoa_noexcept
332  {
333  m_planewaves.push_back(Planewave<D, T>(1ul, 0., HOA_PI2));

341  void generateIcosahedron() hoa_noexcept
342  {
343  m_planewaves.push_back(Planewave<D, T>(1, 0., HOA_PI2));
344  for(size_t i = 1; i < 6; i++)
345  {
346  m_planewaves.push_back(Planewave<D, T>(i*2, static_cast<T>(i - 1.) / 5. * HOA_2PI, atan(0.5)));
347  m_planewaves.push_back(Planewave<D, T>(i*2+1, T(i - 1.) / 5. * HOA_2PI - HOA_PI / 5., -atan(0.5)));
348  }
349  m_planewaves.push_back(Planewave<D, T>(12, 0., static_cast<T>(-HOA_PI2)));
350  }

352  void generateDodecahedron() hoa_noexcept
353  {
354  const T phi = (sqrt(5.) - 1.) / 2.;
355  const T R = 1. / sqrt(3.);
356  const T a = R;
357  const T b = R / phi;
358  const T c = R * phi;
359  size_t index = 1;
360  for(long i = -1; i < 2; i += 2)
361  {
362  for(long j = -1; j < 2; j += 2)
363  {
364  m_planewaves.push_back(Planewave<D, T>(index++, 0., i * c * R, -j * b * R));
365  m_planewaves.push_back(Planewave<D, T>(index++, i * c * R, j * b * R, 0.));
366  m_planewaves.push_back(Planewave<D, T>(index++, i * b * R, 0., -j * c * R));
367  for(long k = -1; k < 2; k += 2)
368  {
369  m_planewaves.push_back(Planewave<D, T>(index++, i * a * R, j * a * R, -k * a * R));
370  }
371  }
372  }
373  }

375  void generatePolyhedron(size_t nplws) hoa_noexcept
376  {
377  if(nplws % 2)
378  {
379  m_planewaves.push_back(Planewave<D, T>(1, 0., static_cast<T>(HOA_PI2)));
380  }
381  const T phi = (sqrt(5.) - 1.) / 4.;
382  const size_t limit = (nplws - (nplws % 2)) / 2;
383  const T offset = 1. / T(limit) * HOA_PI;
384  for(size_t i = 0; i < nplws - (nplws % 2); i++)
385  {
386  if(i < limit)
387  {
388  m_planewaves.push_back(Planewave<D, T>(i+1, T(i) / T(limit) * HOA_2PI, phi));
389  }
390  else
391  {
392  m_planewaves.push_back(Planewave<D, T>(i+1, T(i - (limit)) / T(limit) * HOA_2PI + offset, phi * 2));
393  }
394  }
395  }
396  };
