106  ProcessorPlanewaves(const size_t numberOfPlanewaves) hoa_noexcept :
107  m_rotation_z(0.), m_rotation_y(0.), m_rotation_x(0.)
108  {
109  if(D == Hoa2d)
110  {
111  generateCicle(numberOfPlanewaves);
112  }
113  else
114  {
115  if(numberOfPlanewaves == 4) {
116  generateTetrahedron();

245  void generateTetrahedron() hoa_noexcept
246  {
247  const double oh = -(sqrt(2. / 3.) / sqrt(3. / 8.) - 1.);
248  const double hc = sqrt(1. - oh * oh);
249  const double el = asin(oh / sqrt(hc*hc + oh*oh));
250  m_planewaves.push_back(Planewave<D, T>(1ul, 0., (T)HOA_PI2));
251  m_planewaves.push_back(Planewave<D, T>(2ul, 0., el));
252  m_planewaves.push_back(Planewave<D, T>(3ul, (T)(HOA_2PI / 3.), el));
253  m_planewaves.push_back(Planewave<D, T>(4ul, (T)(HOA_2PI / 1.5), el));
254  }

279  void generateIcosahedron() hoa_noexcept
280  {
281  m_planewaves.push_back(Planewave<D, T>(1, 0., (T)HOA_PI2));
282  for(size_t i = 1; i < 6; i++)
283  {
284  m_planewaves.push_back(Planewave<D, T>(i*2, T(i - 1.) / 5. * HOA_2PI, atan(0.5)));
285  m_planewaves.push_back(Planewave<D, T>(i*2+1, T(i - 1.) / 5. * HOA_2PI - HOA_PI / 5., -atan(0.5)));
286  }
287  m_planewaves.push_back(Planewave<D, T>(12, 0., (T)-HOA_PI2));
288  }

290  void generateDodecahedron() hoa_noexcept
291  {
292  const T phi = (sqrt(5.) - 1.) / 2.;
293  const T R = 1. / sqrt(3.);
294  const T a = R;
295  const T b = R / phi;
296  const T c = R * phi;
297  size_t index = 1;
298  for(long i = -1; i < 2; i += 2)
299  {
300  for(long j = -1; j < 2; j += 2)
301  {
302  m_planewaves.push_back(Planewave<D, T>(index++, 0., i * c * R, -j * b * R));
303  m_planewaves.push_back(Planewave<D, T>(index++, i * c * R, j * b * R, 0.));
304  m_planewaves.push_back(Planewave<D, T>(index++, i * b * R, 0., -j * c * R));
305  for(long k = -1; k < 2; k += 2)
306  {
307  m_planewaves.push_back(Planewave<D, T>(index++, i * a * R, j * a * R, -k * a * R));
308  }
309  }
310  }
311  }

313  void generatePolyhedron(size_t nplws) hoa_noexcept
314  {
315  if(nplws % 2)
316  {
317  m_planewaves.push_back(Planewave<D, T>(1, 0., (T)HOA_PI2));
318  }
319  const T phi = (sqrt(5.) - 1.) / 4.;
320  const size_t limit = (nplws - (nplws % 2)) / 2;
321  const T offset = 1. / T(limit) * HOA_PI;
322  for(size_t i = 0; i < nplws - (nplws % 2); i++)
323  {
324  if(i < limit)
325  {
326  m_planewaves.push_back(Planewave<D, T>(i+1, T(i) / T(limit) * HOA_2PI, phi));
327  }
328  else
329  {
330  m_planewaves.push_back(Planewave<D, T>(i+1, T(i - (limit)) / T(limit) * HOA_2PI + offset, phi * 2));
331  }
332  }
333  }
334  };
