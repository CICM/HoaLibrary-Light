37  Projector(const size_t order, const size_t numberOfPlanewaves) hoa_noexcept :
38  Encoder<Hoa2d, T>(order),

42  const T factor = 1. / (T)(Encoder<Hoa2d, T>::getDecompositionOrder() + 1.);
43  for(size_t i = 0; i < ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves(); i++)
44  {
45  Encoder<Hoa2d, T>::setAzimuth(ProcessorPlanewaves<Hoa2d, T>::getPlanewaveAzimuth(i));
46  Encoder<Hoa2d, T>::process(&factor, m_matrix + i * Encoder<Hoa2d, T>::getNumberOfHarmonics());
47  m_matrix[i * Encoder<Hoa2d, T>::getNumberOfHarmonics()] = factor * 0.5;
48  }
