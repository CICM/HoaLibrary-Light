49  inline void setWidening(T value) hoa_noexcept {
50  m_widening = std::max(std::min(value, T(1)), T(0));
51  const size_t order = ProcessorHarmonics<D, T>::getDecompositionOrder();
52  const T temp = 1 - m_widening;
