116  inline void process(const T* inputs, T* outputs) hoa_noexcept hoa_override
117  {
118  T cos_x = m_cosx;
119  T sin_x = m_sinx;
120  T tcos_x = cos_x;
121 
122  (*outputs++) = (*inputs++);
123  T sig = (*inputs++);
124  (*outputs++) = sin_x * (*inputs) + cos_x * sig;
125  (*outputs++) = cos_x * (*inputs++) - sin_x * sig;
126  for(size_t i = 2; i <= ProcessorHarmonics<Hoa2d, T>::getDecompositionOrder(); i++)
127  {
128  cos_x = tcos_x * m_cosx - sin_x * m_sinx;
129  sin_x = tcos_x * m_sinx + sin_x * m_cosx;
130  tcos_x = cos_x;
131  sig = (*inputs++);
132  (*outputs++) = sin_x * (*inputs) + cos_x * sig;
133  (*outputs++) = cos_x * (*inputs++) - sin_x * sig;
134  }
135  }
