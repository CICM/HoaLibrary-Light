106  static inline void mul(const size_t m, const size_t n, const size_t l, const T* in1, const T* in2, T* output) hoa_noexcept
107  {
108  size_t i, j, k;
109  memset(output, 0, m * n * sizeof(T));
110  T* out = output;
111  for(k = 0; k < l; k++)
112  {
113  out = output;
114  for(i = 0; i < m; i++)
115  {
116  const T g0 = in1[l * i + k];
117  if(g0 != 0)
118  {
119  const T* in = in2+n*k;
120  for(j = n; j; j -= 8, out += 8, in += 8)
121  {
122  const T f0 = in[0] * g0, f1 = in[1] * g0, f2 = in[2] * g0, f3 = in[3] * g0;
123  const T f4 = in[4] * g0, f5 = in[5] * g0, f6 = in[6] * g0, f7 = in[7] * g0;
124  out[0] += f0; out[1] += f1; out[2] += f2; out[3] += f3;
125  out[4] += f4; out[5] += f5; out[6] += f6; out[7] += f7;
126  }
