96  inline void process(const T* input, T* outputs) hoa_noexcept hoa_override
97  {
98  const size_t nencoders = m_encoders.size();
99  const size_t nharmos = ProcessorHarmonics<D, T>::getNumberOfHarmonics();
100  if(m_encoders[0].getMute())
101  {
102  memset(outputs, 0, nharmos * sizeof(T));
103  }
104  else
105  {
106  m_encoders[0].process(input, outputs);
107  }
108 
109  for(size_t i = 1; i < nencoders; i++)
110  {
111  m_encoders[i].process(++input, m_temp);
112  sigadd(nharmos, m_temp, outputs);
113  }
114  }

117  static inline void sigadd(const size_t size, const T* in, T* out) hoa_noexcept
118  {
119  for(size_t i = size>>3; i; --i, in += 8, out += 8)
120  {
121  out[0] += in[0]; out[1] += in[1]; out[2] += in[2]; out[3] += in[3];
122  out[4] += in[4]; out[5] += in[5]; out[6] += in[6]; out[7] += in[7];
123  }
124  for(size_t i = size&7; i; --i, in++, out++)
125  {
126  out[0] += in[0];
127  }
128  }
