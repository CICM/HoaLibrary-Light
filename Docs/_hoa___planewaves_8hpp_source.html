391  inline T getAbscissa(const T x_axe, const T y_axe, const T z_axe) const hoa_noexcept
392  {
393  T x = cos(m_azimuth + HOA_PI2) * cos(m_elevation);
394  T y = sin(m_azimuth + HOA_PI2) * cos(m_elevation);
395  T z = sin(m_elevation);
396 
397  const T cosAngle = cos(x_axe);
398  const T sinAngle = sin(x_axe);
399  T ry = y * cosAngle - z * sinAngle;
400  z = y * sinAngle + z * cosAngle;
401  y = ry;
402 
403  x = x * cos(z_axe) - y * sin(z_axe);
404 
405  return x * cos(y_axe) - z * sin(y_axe);
406  }

436  T rz = y * sinAngle + z * cosAngle;
437  y = ry;
438  z = rz;

441  sinAngle = sin(z_axe);
442  x = x * cosAngle - y * sinAngle;
443
