120  static inline void rotate(Planewave const& p, const T xr, const T yr, const T zr, T& xo, T& yo, T& zo) hoa_noexcept
121  {
122  const T xi = p.getAbscissa();
123  const T yi = p.getOrdinate();
124  const T zi = p.getHeight();
125  // Around x axis
126  const T cosxr = std::cos(xr);
127  const T sinxr = std::sin(xr);
128  // xx
129  const T yx = cosxr * yi + -sinxr * zi;
130  const T zx = sinxr * yi + cosxr * zi;
131 
132  // Around y axis
133  const T cosyr = std::cos(yr);
134  // yy
135  const T sinyr = std::sin(yr);
136  const T xy = cosyr * xi + sinxr * zx;
137  zo = -sinyr * xi + cosyr * zx;
138 
139  // Around z axis
140  const T coszr = std::cos(zr);
141  const T sinzr = std::sin(zr);
142  xo = coszr * xy + -sinzr * yx;
143  yo = sinzr * xy + coszr * yx;
144  // zz
145  }
