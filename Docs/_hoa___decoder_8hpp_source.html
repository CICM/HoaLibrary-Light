329  void computeRendering(const size_t vectorsize = 64) hoa_override
330  {
331  Encoder<Hoa2d, T> encoder(Decoder<Hoa2d, T>::getDecompositionOrder());
332  Signal<T>::clear(Decoder<Hoa2d, T>::getNumberOfPlanewaves() * Decoder<Hoa2d, T>::getNumberOfHarmonics(), m_matrix);
333  T* vector_harmonics = Signal<T>::alloc(Decoder<Hoa2d, T>::getNumberOfHarmonics());
334 
335  if(Decoder<Hoa2d, T>::getNumberOfPlanewaves() == 1)
336  {
337  const size_t nls = size_t(Decoder<Hoa2d, T>::getDecompositionOrder() + 1.);
338  const T factor = 1. / (T)(nls);
339  for(size_t i = 0; i <nls; i++)
340  {
341  encoder.setAzimuth(T(i) * HOA_2PI / T(nls));
342  encoder.process(&factor, vector_harmonics);
343  vector_harmonics[0] = factor * 0.5;
344  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix);
345  }
346  }
347  else
348  {
349  T smallest_distance = (T)HOA_2PI;
350  std::vector<Planewave<Hoa2d, T> > channels;
351  for(size_t i = 0; i < Decoder<Hoa2d, T>::getNumberOfPlanewaves(); i++)
352  {
353  channels.push_back(Planewave<Hoa2d, T>(i, Math<T>::wrap_twopi(Decoder<Hoa2d, T>::getPlanewaveAzimuth(i)), 0.));
354  }
355 
356  sort(channels.begin(), channels.end(), Planewave<Hoa2d, T>::sort_azimuth);
357 
358  {
359  const T current_angle = channels[0].getAzimuth(0., 0., 0.);
360  const T previous_angle = channels[channels.size() - 1].getAzimuth(0., 0., 0.);
361  const T previous_portion= (HOA_2PI - previous_angle) + current_angle;
362  if(smallest_distance > previous_portion)
363  {
364  smallest_distance = previous_portion;
365  }
366  //post("channel %i : %f", (int)channels[0].getIndex(), (float)(channels[0].getAzimuth() / HOA_2PI * 360.f));
367  }
368  for(size_t i = 1; i < channels.size(); i++)
369  {
370  const T current_angle = channels[i].getAzimuth(0., 0., 0.);
371  const T previous_angle = channels[i-1].getAzimuth(0., 0., 0.);
372  const T previous_portion= current_angle - previous_angle;
373  if(smallest_distance > previous_portion)
374  {
375  smallest_distance = previous_portion;
376  }
377  //post("channel %i : %f", (int)channels[i].getIndex(), (float)(channels[i].getAzimuth() / HOA_2PI * 360.f));
378  }
379  //post("");
380 
381  if(smallest_distance > HOA_2PI / T(Decoder<Hoa2d, T>::getNumberOfHarmonics() + 1.))
382  {
383  smallest_distance = HOA_2PI / T(Decoder<Hoa2d, T>::getNumberOfHarmonics() + 1.);
384  }
385  const size_t nvirtual = (size_t)ceil(HOA_2PI / smallest_distance);
386  const T factor = 1. / (T)(nvirtual);
387 
388  //post("number of virtual %i", nvirtual);
389  for(size_t i = 0; i < nvirtual; i++)
390  {
391  const T angle = T(i) / T(nvirtual) * HOA_2PI;
392  //post("virtual %i : %f", (int)i , (float)(angle / HOA_2PI * 360.f));
393  if(angle < channels[0].getAzimuth(0., 0., 0.))
394  {
395  const T portion = (HOA_2PI - channels[channels.size()-1].getAzimuth(0., 0., 0.)) + channels[0].getAzimuth(0., 0., 0.);
396 
397  const T factor1 = (1. - ((channels[0].getAzimuth(0., 0., 0.) - angle) / portion)) * factor;
398  encoder.setAzimuth(angle);
399  encoder.process(&factor1, vector_harmonics);
400  vector_harmonics[0] = factor1 * 0.5;
401  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[0].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
402 
403  const T factor2 = ((channels[0].getAzimuth(0., 0., 0.) - angle) / portion) * factor;
404  encoder.process(&factor2, vector_harmonics);
405  vector_harmonics[0] = factor2 * 0.5;
406  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[channels.size() - 1].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
407 
408  //post("portion : %f", (float)portion / HOA_2PI * 360.f);
409  //post("channel %i (%f) : %f", (int)channels[channels.size()-1].getIndex(),
410  // (float)(channels[channels.size()-1].getAzimuth() / HOA_2PI * 360.f),
411  // (float)(factor2 / factor) * 100.f);
412  //post("channel %i (%f) : %f", (int)channels[0].getIndex(),
413  // (float)(channels[0].getAzimuth() / HOA_2PI * 360.f),
414  // (float)(factor1 / factor) * 100.f);
415  }
416  else if(angle >= channels[channels.size() - 1].getAzimuth(0., 0., 0.))
417  {
418  const T portion = (HOA_2PI - channels[channels.size()-1].getAzimuth(0., 0., 0.)) + channels[0].getAzimuth(0., 0., 0.);
419 
420  const T factor1 = (1. - ((angle - channels[channels.size()-1].getAzimuth(0., 0., 0.)) / portion)) * factor;
421  encoder.setAzimuth(angle);
422  encoder.process(&factor1, vector_harmonics);
423  vector_harmonics[0] = factor1 * 0.5;
424  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[channels.size()-1].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
425 
426  const T factor2 = ((angle - channels[channels.size()-1].getAzimuth(0., 0., 0.)) / portion) * factor;
427  encoder.process(&factor2, vector_harmonics);
428  vector_harmonics[0] = factor2 * 0.5;
429  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[0].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
430 
431  //post("portion : %f", (float)portion / HOA_2PI * 360.f);
432  //post("channel %i (%f) : %f", (int)channels[channels.size()-1].getIndex(),
433  // (float)(channels[channels.size()-1].getAzimuth() / HOA_2PI * 360.f),
434  // (float)(factor1 / factor) * 100.f);
435  //post("channel %i (%f) : %f", (int)channels[0].getIndex(),
436  // (float)(channels[0].getAzimuth() / HOA_2PI * 360.f),
437  // (float)(factor2 / factor) * 100.f);
438  }
439  else
440  {
441  for(size_t j = 1; j < channels.size(); j++)
442  {
443  if(angle < channels[j].getAzimuth(0., 0., 0.) && angle >= channels[j-1].getAzimuth(0., 0., 0.))
444  {
445  const T portion = (channels[j].getAzimuth(0., 0., 0.) - channels[j-1].getAzimuth(0., 0., 0.));
446 
447  const T factor1 = (1. - ((channels[j].getAzimuth(0., 0., 0.) - angle) / portion)) * factor;
448  encoder.setAzimuth(angle);
449  encoder.process(&factor1, vector_harmonics);
450  vector_harmonics[0] = factor1 * 0.5;
451  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[j].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
452 
453  const T factor2 = ((channels[j].getAzimuth(0., 0., 0.) - angle) / portion) * factor;
454  encoder.process(&factor2, vector_harmonics);
455  vector_harmonics[0] = factor2 * 0.5;
456  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[j-1].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
457 
458  //post("portion : %f", (float)portion / HOA_2PI * 360.f);
459  //post("channel %i (%f) : %f", (int)channels[j-1].getIndex(),
460  // (float)(channels[j-1].getAzimuth() / HOA_2PI * 360.f),
461  // (float)(factor2 / factor) * 100.f);
462  //post("channel %i (%f) : %f", (int)channels[j].getIndex(),
463  // (float)(channels[j].getAzimuth() / HOA_2PI * 360.f),
464  // (float)(factor1 / factor) * 100.f);
465 
466  break;
467  }
468  }
469  }
470  //post("");
471  }
472  channels.clear();
473  }
474  Signal<T>::free(vector_harmonics);
475  }

565  m_vector_size = vectorsize;

573  inline void processChannel(const T* harmonics, const T* response, T* vector, T* output) hoa_noexcept
574  {
575  const size_t l = Hrir<Hoa2d, T>::getNumberOfColumns(); // Harmonics size aka 11
576  const size_t m = m_crop_size; // Impulses size
577  const size_t n = m_vector_size; // Vector size
578  Signal<T>::mul(m, n, l, response, harmonics, m_result);
579  for(size_t i = 0; i < n; i ++)
580  {
581  Signal<T>::add(m, m_result + i, n, vector + i, 1ul);
582  }
583  Signal<T>::copy(m_vector_size, vector, output);
584  Signal<T>::copy(m, vector + m_vector_size, vector);
585  Signal<T>::clear(m_vector_size, vector + m);

595  Signal<T>::copy(m_vector_size, inputs[i], input);
596  input += m_vector_size;
597  }
598  processChannel(m_input, Hrir<Hoa2d, T>::getLeftMatrix(), m_left, outputs[0]);
599  processChannel(m_input, Hrir<Hoa2d, T>::getRightMatrix(), m_right, outputs[1]);
600  }

722  for(size_t i = 0; i < Decoder<Hoa3d, T>::getNumberOfPlanewaves(); i++)
723  {
