81  DecoderRegular(size_t order, size_t nplws) : Decoder<D, T>(order, nplws)
82  {
83  m_matrix = new T[Decoder<D, T>::getNumberOfPlanewaves() * Decoder<D, T>::getNumberOfHarmonics()];
84  prepare();
85  }

88  ~DecoderRegular() {
89  delete [] m_matrix;

98  void process(const T* inputs, T* outputs) hoa_noexcept hoa_override
99  {
100  const size_t nharm = Decoder<D, T>::getNumberOfHarmonics();
101  const size_t nplws = Decoder<D, T>::getNumberOfPlanewaves();
102  const T* matrix = m_matrix;
103  for(size_t i = 0ul; i < nplws; i++)
104  {
105  T result = 0;
106  const T* in1 = inputs;
107  for(size_t j = nharm>>3; j; --j, in1 += 8, matrix += 8)
108  {
109  result += in1[0] * matrix[0]; result += in1[1] * matrix[1];
110  result += in1[2] * matrix[2]; result += in1[3] * matrix[3];
111  result += in1[4] * matrix[4]; result += in1[5] * matrix[5];
112  result += in1[6] * matrix[6]; result += in1[7] * matrix[7];
113  }
114  for(size_t j = nharm&7; j; --j, in1++, matrix++)
115  {
116  result += in1[0] * matrix[0];
117  }
118  outputs[i] = result;
119  }
120  }
121 
123  void prepare(const size_t vectorsize = 64) hoa_override
124  {
125  const size_t order = Decoder<D, T>::getDecompositionOrder();
126  const size_t nharm = Decoder<D, T>::getNumberOfHarmonics();
127  const size_t nplws = Decoder<D, T>::getNumberOfPlanewaves();
128 
129  Encoder<D, T> encoder(order);
130  const T factor = T(1.) / T(order * 2 + 1);// / ((D == Hoa2d) ? T(order + 1) : T(1.));
131  for(size_t i = 0; i < nplws; i++)
132  {
133  encoder.setAzimuth(Decoder<D, T>::getPlanewaveAzimuth(i));
134  encoder.setElevation(Decoder<D, T>::getPlanewaveElevation(i));
135  encoder.process(&factor, m_matrix + i * nharm);
136  for(size_t j = 0; j < nharm; ++j)
137  {
138  const size_t degree = Decoder<D, T>::getHarmonicDegree(j);
139  m_matrix[i * nharm + j] *= (T(2) * T(degree) + T(1));
140  }
141  }
142  }

145  inline typename Decoder<D, T>::Mode getMode() const hoa_noexcept {return Decoder<D, T>::RegularMode;}

332  void prepare(const size_t vectorsize = 64) hoa_override
333  {
334  Encoder<Hoa2d, T> encoder(Decoder<Hoa2d, T>::getDecompositionOrder());
335  Signal<T>::clear(Decoder<Hoa2d, T>::getNumberOfPlanewaves() * Decoder<Hoa2d, T>::getNumberOfHarmonics(), m_matrix);
336  T* vector_harmonics = Signal<T>::alloc(Decoder<Hoa2d, T>::getNumberOfHarmonics());
337 
338  if(Decoder<Hoa2d, T>::getNumberOfPlanewaves() == 1)
339  {
340  const size_t nls = size_t(Decoder<Hoa2d, T>::getDecompositionOrder() + 1.);
341  const T factor = 1. / (T)(nls);
342  for(size_t i = 0; i <nls; i++)
343  {
344  encoder.setAzimuth(T(i) * HOA_2PI / T(nls));
345  encoder.process(&factor, vector_harmonics);
346  vector_harmonics[0] = factor * 0.5;
347  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix);
348  }
349  }
350  else
351  {
352  T smallest_distance = (T)HOA_2PI;
353  std::vector<Planewave<Hoa2d, T> > channels;
354  for(size_t i = 0; i < Decoder<Hoa2d, T>::getNumberOfPlanewaves(); i++)
355  {
356  T nazimtuh = Decoder<Hoa2d, T>::getPlanewaveAzimuth(i);
357  while(nazimtuh < T(0.)) {
358  nazimtuh += T(HOA_2PI);
359  }
360  while(nazimtuh >= T(HOA_2PI)) {
361  nazimtuh -= T(HOA_2PI);
362  }
363  channels.push_back(Planewave<Hoa2d, T>(i, nazimtuh, 0.));
364  }
365 
366  sort(channels.begin(), channels.end(), Planewave<Hoa2d, T>::compare_azimuth);
367 
368  {
369  const T current_angle = channels[0].getAzimuth(0., 0., 0.);
370  const T previous_angle = channels[channels.size() - 1].getAzimuth(0., 0., 0.);
371  const T previous_portion= (HOA_2PI - previous_angle) + current_angle;
372  if(smallest_distance > previous_portion)
373  {
374  smallest_distance = previous_portion;
375  }
376  //post("channel %i : %f", (int)channels[0].getIndex(), (float)(channels[0].getAzimuth() / HOA_2PI * 360.f));
377  }
378  for(size_t i = 1; i < channels.size(); i++)
379  {
380  const T current_angle = channels[i].getAzimuth(0., 0., 0.);
381  const T previous_angle = channels[i-1].getAzimuth(0., 0., 0.);
382  const T previous_portion= current_angle - previous_angle;
383  if(smallest_distance > previous_portion)
384  {
385  smallest_distance = previous_portion;
386  }
387  //post("channel %i : %f", (int)channels[i].getIndex(), (float)(channels[i].getAzimuth() / HOA_2PI * 360.f));
388  }
389  //post("");
390 
391  if(smallest_distance > HOA_2PI / T(Decoder<Hoa2d, T>::getNumberOfHarmonics() + 1.))
392  {
393  smallest_distance = HOA_2PI / T(Decoder<Hoa2d, T>::getNumberOfHarmonics() + 1.);
394  }
395  const size_t nvirtual = (size_t)ceil(HOA_2PI / smallest_distance);
396  const T factor = 1. / (T)(nvirtual);
397 
398  //post("number of virtual %i", nvirtual);
399  for(size_t i = 0; i < nvirtual; i++)
400  {
401  const T angle = T(i) / T(nvirtual) * HOA_2PI;
402  //post("virtual %i : %f", (int)i , (float)(angle / HOA_2PI * 360.f));
403  if(angle < channels[0].getAzimuth(0., 0., 0.))
404  {
405  const T portion = (HOA_2PI - channels[channels.size()-1].getAzimuth(0., 0., 0.)) + channels[0].getAzimuth(0., 0., 0.);
406 
407  const T factor1 = (1. - ((channels[0].getAzimuth(0., 0., 0.) - angle) / portion)) * factor;
408  encoder.setAzimuth(angle);
409  encoder.process(&factor1, vector_harmonics);
410  vector_harmonics[0] = factor1 * 0.5;
411  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[0].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
412 
413  const T factor2 = ((channels[0].getAzimuth(0., 0., 0.) - angle) / portion) * factor;
414  encoder.process(&factor2, vector_harmonics);
415  vector_harmonics[0] = factor2 * 0.5;
416  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[channels.size() - 1].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
417 
418  //post("portion : %f", (float)portion / HOA_2PI * 360.f);
419  //post("channel %i (%f) : %f", (int)channels[channels.size()-1].getIndex(),
420  // (float)(channels[channels.size()-1].getAzimuth() / HOA_2PI * 360.f),
421  // (float)(factor2 / factor) * 100.f);
422  //post("channel %i (%f) : %f", (int)channels[0].getIndex(),
423  // (float)(channels[0].getAzimuth() / HOA_2PI * 360.f),
424  // (float)(factor1 / factor) * 100.f);
425  }
426  else if(angle >= channels[channels.size() - 1].getAzimuth(0., 0., 0.))
427  {
428  const T portion = (HOA_2PI - channels[channels.size()-1].getAzimuth(0., 0., 0.)) + channels[0].getAzimuth(0., 0., 0.);
429 
430  const T factor1 = (1. - ((angle - channels[channels.size()-1].getAzimuth(0., 0., 0.)) / portion)) * factor;
431  encoder.setAzimuth(angle);
432  encoder.process(&factor1, vector_harmonics);
433  vector_harmonics[0] = factor1 * 0.5;
434  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[channels.size()-1].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
435 
436  const T factor2 = ((angle - channels[channels.size()-1].getAzimuth(0., 0., 0.)) / portion) * factor;
437  encoder.process(&factor2, vector_harmonics);
438  vector_harmonics[0] = factor2 * 0.5;
439  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[0].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
440 
441  //post("portion : %f", (float)portion / HOA_2PI * 360.f);
442  //post("channel %i (%f) : %f", (int)channels[channels.size()-1].getIndex(),
443  // (float)(channels[channels.size()-1].getAzimuth() / HOA_2PI * 360.f),
444  // (float)(factor1 / factor) * 100.f);
445  //post("channel %i (%f) : %f", (int)channels[0].getIndex(),
446  // (float)(channels[0].getAzimuth() / HOA_2PI * 360.f),
447  // (float)(factor2 / factor) * 100.f);
448  }
449  else
450  {
451  for(size_t j = 1; j < channels.size(); j++)
452  {
453  if(angle < channels[j].getAzimuth(0., 0., 0.) && angle >= channels[j-1].getAzimuth(0., 0., 0.))
454  {
455  const T portion = (channels[j].getAzimuth(0., 0., 0.) - channels[j-1].getAzimuth(0., 0., 0.));
456 
457  const T factor1 = (1. - ((channels[j].getAzimuth(0., 0., 0.) - angle) / portion)) * factor;
458  encoder.setAzimuth(angle);
459  encoder.process(&factor1, vector_harmonics);
460  vector_harmonics[0] = factor1 * 0.5;
461  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[j].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
462 
463  const T factor2 = ((channels[j].getAzimuth(0., 0., 0.) - angle) / portion) * factor;
464  encoder.process(&factor2, vector_harmonics);
465  vector_harmonics[0] = factor2 * 0.5;
466  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[j-1].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
467 
468  //post("portion : %f", (float)portion / HOA_2PI * 360.f);
469  //post("channel %i (%f) : %f", (int)channels[j-1].getIndex(),
470  // (float)(channels[j-1].getAzimuth() / HOA_2PI * 360.f),
471  // (float)(factor2 / factor) * 100.f);
472  //post("channel %i (%f) : %f", (int)channels[j].getIndex(),
473  // (float)(channels[j].getAzimuth() / HOA_2PI * 360.f),
474  // (float)(factor1 / factor) * 100.f);
475 
476  break;
477  }
478  }
479  }
480  //post("");
481  }
482  channels.clear();
483  }
484  Signal<T>::free(vector_harmonics);
485  }

586  const size_t m = m_crop_size; // Impulses size
587  const size_t n = m_vector_size; // Vector size

591  Signal<T>::add(m, m_result + i, n, vector + i, 1ul);
592  }
593  Signal<T>::copy(m_vector_size, vector, output);
594  Signal<T>::copy(m, vector + m_vector_size, vector);

606  input += m_vector_size;
607  }
608  processChannel(m_input, Hrir<Hoa2d, T>::getLeftMatrix(), m_left, outputs[0]);
609  processChannel(m_input, Hrir<Hoa2d, T>::getRightMatrix(), m_right, outputs[1]);
610  }
