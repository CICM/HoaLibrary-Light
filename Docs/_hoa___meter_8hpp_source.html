90  void computeRendering()
91  {
92  if(ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves() == 1)
93  {
94  m_channels_azimuth_width[0] = HOA_2PI;
95  m_channels_azimuth_mapped[0]= 0.;
96  }
97  else
98  {
99  std::vector<Planewave<Hoa2d, T> > channels;
100  for(size_t i = 0; i < ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves(); i++)
101  {
102  channels.push_back(Planewave<Hoa2d, T>(i, Math<T>::wrap_twopi(ProcessorPlanewaves<Hoa2d, T>::getPlanewaveAzimuth(i)), 0.));
103  }
104  std::sort(channels.begin(), channels.end(), Planewave<Hoa2d, T>::sort_azimuth);
105  {
106  const T current_angle = channels[0].getAzimuth(0., 0., 0.);
107  const T previous_angle = channels[channels.size() - 1].getAzimuth(0., 0., 0.);
108  const T next_angle = channels[1].getAzimuth(0., 0., 0.);
109  const T previous_portion= (HOA_2PI - previous_angle) + current_angle;
110  const T next_portion = next_angle - current_angle;
111  m_channels_azimuth_width[channels[0].getIndex()] = (previous_portion + next_portion) * 0.5;
112  m_channels_azimuth_mapped[channels[0].getIndex()]= Math<T>::wrap_twopi((current_angle - previous_portion * 0.5) + m_channels_azimuth_width[channels[0].getIndex()] * 0.5);
113  }
114  for(size_t i = 1; i < channels.size() - 1; i++)
115  {
116  const T current_angle = channels[i].getAzimuth(0., 0., 0.);
117  const T previous_angle = channels[i-1].getAzimuth(0., 0., 0.);
118  const T next_angle = channels[i+1].getAzimuth(0., 0., 0.);
119  const T previous_portion= current_angle - previous_angle;
120  const T next_portion = next_angle - current_angle;
121  m_channels_azimuth_width[channels[i].getIndex()] = (previous_portion + next_portion) * 0.5;
122  m_channels_azimuth_mapped[channels[i].getIndex()]= Math<T>::wrap_twopi((current_angle - previous_portion * 0.5) + m_channels_azimuth_width[channels[i].getIndex()] * 0.5);
123  }
124  {
125  const size_t index = channels.size() - 1;
126  const T current_angle = channels[index].getAzimuth(0., 0., 0.);
127  const T previous_angle = channels[index - 1].getAzimuth(0., 0., 0.);
128  const T next_angle = channels[0].getAzimuth(0., 0., 0.);
129  const T previous_portion= current_angle - previous_angle;
130  const T next_portion = (HOA_2PI - current_angle) + next_angle;
131  m_channels_azimuth_width[channels[index].getIndex()] = (previous_portion + next_portion) * 0.5;
132  m_channels_azimuth_mapped[channels[index].getIndex()]= Math<T>::wrap_twopi((current_angle - previous_portion * 0.5) + m_channels_azimuth_width[channels[index].getIndex()] * 0.5);
133  }
134  channels.clear();
135  }
136  }
