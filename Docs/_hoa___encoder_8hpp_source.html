124  void setRadius(const T radius) hoa_noexcept {
125  m_radius = std::max(radius, static_cast<T>(0));
126 
127  if(m_radius >= 1)
128  {
129  const size_t order = ProcessorHarmonics<D, T>::getDecompositionOrder();
130  const T gain = static_cast<T>(1) / m_radius;
131  T* coeff = m_radius_coeffs.data();
132  (*coeff++) = gain;
133  for(size_t i = 1; i <= order; ++i) {
134  (*coeff++) = gain;
135  }
136  }
137  else
138  {
139  const size_t order = ProcessorHarmonics<D, T>::getDecompositionOrder();
140  const T widening = m_radius;
141  const T temp = 1 - widening;
142  T* coeff = m_radius_coeffs.data();
143  (*coeff++) = static_cast<T>(order) * temp + static_cast<T>(1);
144  for(size_t i = 1; i <= order; ++i) {
145  (*coeff++) = std::pow(widening, static_cast<T>(i)) * (temp * static_cast<T>(order - i) + static_cast<T>(1.));
146  }
147  }
148 
149  }

153  void setAzimuth(const T azimuth) hoa_noexcept {
154  m_azimuth = azimuth;
155  T* coeffs = m_azimuth_coeffs.data();
156  const size_t order = ProcessorHarmonics<D, T>::getDecompositionOrder();
157  T const ccos_x = std::cos(azimuth);
158  T const csin_x = std::sin(azimuth);
159  if(D == Hoa2d)
160  {
161  // Organization 0, -1, 1, -2, 2, -3, 3,...
162  T cos_x = ccos_x;
163  T sin_x = csin_x;
164  T tcos_x = cos_x;
165  (*coeffs++) = 1; // Hamonic [0]
166  (*coeffs++) = sin_x; // Hamonic [-1]
167  (*coeffs++) = cos_x; // Hamonic [1]
168  for(size_t i = 2; i <= order; i++)
169  {
170  cos_x = tcos_x * ccos_x - sin_x * csin_x;
171  sin_x = tcos_x * csin_x + sin_x * ccos_x;
172  tcos_x = cos_x;
173  (*coeffs++) = sin_x; // Hamonic [l,-l]
174  (*coeffs++) = cos_x; // Hamonic [l,l]
175  }
176  }
177  else
178  {
179  // Organization ..., -3, -2, -1, 0, 1, 2, 3, ...
180  T cos_x = ccos_x;
181  T sin_x = csin_x;
182  T tcos_x = cos_x;
183  coeffs[order] = 1; // Hamonic [0]
184  coeffs[order-1] = sin_x; // Hamonic [-1]
185  coeffs[order+1] = cos_x; // Hamonic [1]
186  for(size_t i = 2; i <= order; i++)
187  {
188  cos_x = tcos_x * ccos_x - sin_x * csin_x;
189  sin_x = tcos_x * csin_x + sin_x * ccos_x;
190  tcos_x = cos_x;
191  coeffs[order-i] = sin_x; // Hamonic [l,-l]
192  coeffs[order+i] = cos_x; // Hamonic [l,l]
193  }
194  }
195  }

199  void setElevation(const T elevation) hoa_noexcept {
200  m_elevation = elevation;
201  T* coeffs = m_elevation_coeffs.data();
202  const size_t order = ProcessorHarmonics<D, T>::getDecompositionOrder();
203  const T sin_theta = std::sin(elevation);
204  const T sqr_theta = -std::sqrt(static_cast<T>(1) - sin_theta * sin_theta);
205 
206  // Organization [0, 0], [1, 1], [1, 0], [2, 2], [2, 1], [2, 0], [3, 3],
207  // [3, 2], [3, 1], [3, 0], [4, 4], ...
208 
209  *(coeffs++) = 1; // P(0, 0)(x) = 1
210  *(coeffs++) = sqr_theta; // P(1, 1)(x) = -(2*0+1)sqrt(1-x^2)P(0,0)(x) = -sqrt(1-x^2)
211  *(coeffs++) = sin_theta; // P(1, 0)(x) = x(2*0+1)P(0,0)(x) = x
212  for(size_t i = 2; i <= order; ++i)
213  {
214  const T ratio = 2 * static_cast<T>(i-1) + 1;
215  const T previous = *(coeffs-i);
216  // P(l+1,l+1)(x) = -(2l+1) sqrt(1-x^2) P(l,l)(x)
217  *(coeffs++) = ratio * sqr_theta * previous;
218  // P(l+1,l)(x) = x(2l+1) P(l,l)(x)
219  *(coeffs++) = ratio * sin_theta * previous;
220  for(size_t j = i - 2; j > 0; --j)
221  {
222  const T previous1 = *(coeffs-(i+1));
223  const T previous2 = *(coeffs-(2*i+1));
224  // P(l+1,m)(x) = (x(2l+1)P(l,m)(x) - (l+m)P(l-1, m)(x)) / (l-m+1)
225  *(coeffs++) = ((ratio * sin_theta * previous1) - static_cast<T>(i-1+j) * previous2) / static_cast<T>(i-j);
226  }
227  const T previous1 = *(coeffs-(i+1));
228  const T previous2 = *(coeffs-(2*i+1));
229  // P(l+1,0)(x) = (x(2l+1)P(l,0)(x) - (l)P(l-1, 0)(x)) / (l+1)
230  *(coeffs++) = ((ratio * sin_theta * previous1) - static_cast<T>(i-1) * previous2) / static_cast<T>(i);
231  }
232  }

240  void process(const T* input, T* outputs) hoa_noexcept hoa_override
241  {
242  if(D == Hoa2d)
243  {
244  const size_t order = ProcessorHarmonics<D, T>::getDecompositionOrder();
245  T const* radius_coeffs = m_radius_coeffs.data();
246  T const* azimuth_coeffs = m_azimuth_coeffs.data();
247  (*outputs++) = (*input) * (*radius_coeffs++) * (*azimuth_coeffs++);
248  for(size_t i = 1; i <= order; ++i)
249  {
250  const T factor = (*radius_coeffs++);
251  (*outputs++) = (*input) * factor * (*azimuth_coeffs++);
252  (*outputs++) = (*input) * factor * (*azimuth_coeffs++);
253  }
254  }
255  else
256  {
257  const size_t order = ProcessorHarmonics<D, T>::getDecompositionOrder();
258  T const* radius_coeffs = m_radius_coeffs.data();
259  T const* azimuth_coeffs = m_azimuth_coeffs.data()+order;
260  T const* elevation_coeffs = m_elevation_coeffs.data();
261  T const* norm_coeffs = m_normalization_coeffs.data();
262  (*outputs++) = (*input) * (*radius_coeffs++) * (*azimuth_coeffs--) * (*elevation_coeffs++) * (*norm_coeffs++);
263  for(size_t i = 1; i <= order; ++i)
264  {
265  const T factor = (*radius_coeffs++);
266  T const* az_coeffs2 = azimuth_coeffs--;
267  T const* el_coeffs2 = elevation_coeffs;
268  elevation_coeffs = elevation_coeffs+(i+1);
269  for(size_t j = i; j; --j)
270  {
271  //(*outputs++) = (*el_coeffs2++);
272  (*outputs++) = (*input) * factor * (*az_coeffs2++) * (*el_coeffs2++) * (*norm_coeffs++);
273  }
274  //(*outputs++) = (*el_coeffs2--);
275  (*outputs++) = (*input) * factor * (*az_coeffs2++) * (*el_coeffs2--) * (*norm_coeffs++);
276  for(size_t j = i; j; --j)
277  {
278  //(*outputs++) = (*el_coeffs2--);
279  (*outputs++) = (*input) * factor * (*az_coeffs2++) * (*el_coeffs2--) * (*norm_coeffs++);
280  }
281  }
282  }

296  static inline T wrap_pi(T value)
297  {
298  while(value < static_cast<T>(-HOA_PI)) {
299  value += static_cast<T>(HOA_2PI);
300  }
301  while(value >= static_cast<T>(HOA_PI)) {
302  value -= static_cast<T>(HOA_2PI);
303  }
304  return value;
