117  void computeRendering() hoa_noexcept
118  {
119  const T factor = 1. / (T)(Encoder<Hoa2d, T>::getDecompositionOrder() + 1.);
120  for(size_t i = 0; i < ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves(); i++)
121  {
122  Encoder<Hoa2d, T>::Basic::setAzimuth(ProcessorPlanewaves<Hoa2d, T>::getPlanewaveAzimuth(i));
123  Encoder<Hoa2d, T>::Basic::process(&factor, m_matrix + i * Encoder<Hoa2d, T>::getNumberOfHarmonics());
124  m_matrix[i * Encoder<Hoa2d, T>::getNumberOfHarmonics()] = factor * 0.5;
125  }
126  for(size_t i = 0; i < ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves(); i++)
127  {
128  m_vector[i] = 0.;
129  }
130  m_maximum = 0;

244  Scope(size_t order, size_t numberOfRow, size_t numberOfColumn) hoa_noexcept :
245  Encoder<Hoa3d, T>::Basic(order),
246  ProcessorPlanewaves<Hoa3d, T>(numberOfRow * numberOfColumn),

249  {
250  for(size_t i = 0; i < m_number_of_rows; i++)
251  {
252  const T elevation = (T)i * HOA_PI / (T)(m_number_of_rows - 1) - HOA_PI2;
253  for(size_t j = 0; j < m_number_of_columns; j++)
254  {
255  ProcessorPlanewaves<Hoa3d, T>::setPlanewaveAzimuth(i * m_number_of_columns + j, (T)j * HOA_2PI / (T)m_number_of_columns);
256  ProcessorPlanewaves<Hoa3d, T>::setPlanewaveElevation(i * m_number_of_columns + j, elevation);
257  }

389  void computeRendering() hoa_noexcept
390  {
391  const T factor = 12.5 / (T)(Encoder<Hoa3d, T>::getNumberOfHarmonics());
392  for(size_t i = 0; i < ProcessorPlanewaves<Hoa3d, T>::getNumberOfPlanewaves(); i++)
393  {
394  Encoder<Hoa3d, T>::Basic::setAzimuth(ProcessorPlanewaves<Hoa3d, T>::getPlanewaveAzimuth(i));
395  Encoder<Hoa3d, T>::Basic::setElevation(ProcessorPlanewaves<Hoa3d, T>::getPlanewaveElevation(i));
396  Encoder<Hoa3d, T>::Basic::process(&factor, m_matrix + i * Encoder<Hoa3d, T>::getNumberOfHarmonics());
397  for(size_t j = 0; j < Encoder<Hoa3d, T>::getNumberOfHarmonics(); j++)
398  {
399  const size_t l = Encoder<Hoa3d, T>::getHarmonicDegree(j);
400  if(Encoder<Hoa3d, T>::getHarmonicOrder(j) == 0)
401  {
402  m_matrix[i * Encoder<Hoa3d, T>::getNumberOfHarmonics() + j] *= (2. * l + 1.);
403  }
404  else
405  {
406  m_matrix[i * Encoder<Hoa3d, T>::getNumberOfHarmonics() + j] *= T(2. * l + 1.) * 4. * HOA_PI;
407  }
408  }
409  }
410  for(size_t i = 0; i < ProcessorPlanewaves<Hoa3d, T>::getNumberOfPlanewaves(); i++)
411  {
412  m_vector[i] = 0.;
413  }
414  m_maximum = 0;
415  }
