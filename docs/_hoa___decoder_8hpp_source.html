96  void process(const T* inputs, T* outputs) hoa_noexcept hoa_override
97  {
98  const size_t nharm = Decoder<D, T>::getNumberOfHarmonics();
99  const size_t nplws = Decoder<D, T>::getNumberOfPlanewaves();
100  const T* matrix = m_matrix.data();
101  for(size_t i = 0ul; i < nplws; i++)
102  {
103  T result = 0;
104  const T* in1 = inputs;
105  for(size_t j = nharm>>3; j; --j, in1 += 8, matrix += 8)
106  {
107  result += in1[0] * matrix[0]; result += in1[1] * matrix[1];
108  result += in1[2] * matrix[2]; result += in1[3] * matrix[3];
109  result += in1[4] * matrix[4]; result += in1[5] * matrix[5];
110  result += in1[6] * matrix[6]; result += in1[7] * matrix[7];
111  }
112  for(size_t j = nharm&7; j; --j, in1++, matrix++)
113  {
114  result += in1[0] * matrix[0];
115  }
116  outputs[i] = result;
117  }
118  }
119 
121  void prepare(const size_t vectorsize = 64) hoa_override
122  {
123  const size_t order = Decoder<D, T>::getDecompositionOrder();
124  const size_t nharm = Decoder<D, T>::getNumberOfHarmonics();
125  const size_t nplws = Decoder<D, T>::getNumberOfPlanewaves();
126 
127  Encoder<D, T> encoder(order);
128  const T factor = T(1.) / T(order * 2 + 1);// / ((D == Hoa2d) ? T(order + 1) : T(1.));
129  for(size_t i = 0; i < nplws; i++)
130  {
131  encoder.setAzimuth(Decoder<D, T>::getPlanewaveAzimuth(i));
132  encoder.setElevation(Decoder<D, T>::getPlanewaveElevation(i));
133  encoder.process(&factor, m_matrix.data() + i * nharm);
134  if(D == Hoa2d)
135  {
136  m_matrix[i * nharm] *= T(0.5);
137  }
138  else
139  {
140  for(size_t j = 0; j < nharm; ++j)
141  {
142  const size_t degree = Decoder<D, T>::getHarmonicDegree(j);
143  m_matrix[i * nharm + j] *= (T(2) * T(degree) + T(1));
144  }
145  }
146  }
147  }

337  void prepare(const size_t vectorsize = 64) hoa_override
338  {
339  Encoder<Hoa2d, T> encoder(Decoder<Hoa2d, T>::getDecompositionOrder());
340  Signal<T>::clear(Decoder<Hoa2d, T>::getNumberOfPlanewaves() * Decoder<Hoa2d, T>::getNumberOfHarmonics(), m_matrix);
341  T* vector_harmonics = Signal<T>::alloc(Decoder<Hoa2d, T>::getNumberOfHarmonics());
342 
343  if(Decoder<Hoa2d, T>::getNumberOfPlanewaves() == 1)
344  {
345  const size_t nls = size_t(Decoder<Hoa2d, T>::getDecompositionOrder() + 1.);
346  const T factor = static_cast<T>(1.) / static_cast<T>(nls);
347  for(size_t i = 0; i <nls; i++)
348  {
349  encoder.setAzimuth(T(i) * HOA_2PI / T(nls));
350  encoder.process(&factor, vector_harmonics);
351  vector_harmonics[0] = factor * 0.5;
352  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix);
353  }
354  }
355  else
356  {
357  T smallest_distance = static_cast<T>(HOA_2PI);
358  std::vector<Planewave<Hoa2d, T> > channels;
359  for(size_t i = 0; i < Decoder<Hoa2d, T>::getNumberOfPlanewaves(); i++)
360  {
361  T nazimtuh = Decoder<Hoa2d, T>::getPlanewaveAzimuth(i);
362  while(nazimtuh < T(0.)) {
363  nazimtuh += T(HOA_2PI);
364  }
365  while(nazimtuh >= T(HOA_2PI)) {
366  nazimtuh -= T(HOA_2PI);
367  }
368  channels.push_back(Planewave<Hoa2d, T>(i, nazimtuh, 0.));
369  }
370 
371  sort(channels.begin(), channels.end(), Planewave<Hoa2d, T>::compare_azimuth);
372 
373  {
374  const T current_angle = channels[0].getAzimuth();
375  const T previous_angle = channels[channels.size() - 1].getAzimuth();
376  const T previous_portion= (HOA_2PI - previous_angle) + current_angle;
377  if(smallest_distance > previous_portion)
378  {
379  smallest_distance = previous_portion;
380  }
381  //post("channel %i : %f", (int)channels[0].getIndex(), (float)(channels[0].getAzimuth() / HOA_2PI * 360.f));
382  }
383  for(size_t i = 1; i < channels.size(); i++)
384  {
385  const T current_angle = channels[i].getAzimuth();
386  const T previous_angle = channels[i-1].getAzimuth();
387  const T previous_portion= current_angle - previous_angle;
388  if(smallest_distance > previous_portion)
389  {
390  smallest_distance = previous_portion;
391  }
392  //post("channel %i : %f", (int)channels[i].getIndex(), (float)(channels[i].getAzimuth() / HOA_2PI * 360.f));
393  }
394  //post("");
395 
396  if(smallest_distance > HOA_2PI / T(Decoder<Hoa2d, T>::getNumberOfHarmonics() + 1.))
397  {
398  smallest_distance = HOA_2PI / T(Decoder<Hoa2d, T>::getNumberOfHarmonics() + 1.);
399  }
400  const size_t nvirtual = static_cast<size_t>(std::ceil(static_cast<T>(HOA_2PI) / smallest_distance));
401  const T factor = static_cast<T>(1) / static_cast<T>(nvirtual);
402 
403  //post("number of virtual %i", nvirtual);
404  for(size_t i = 0; i < nvirtual; i++)
405  {
406  const T angle = T(i) / T(nvirtual) * HOA_2PI;
407  //post("virtual %i : %f", (int)i , (float)(angle / HOA_2PI * 360.f));
408  if(angle < channels[0].getAzimuth())
409  {
410  const T portion = (HOA_2PI - channels[channels.size()-1].getAzimuth()) + channels[0].getAzimuth();
411 
412  const T factor1 = (1. - ((channels[0].getAzimuth() - angle) / portion)) * factor;
413  encoder.setAzimuth(angle);
414  encoder.process(&factor1, vector_harmonics);
415  vector_harmonics[0] = factor1 * 0.5;
416  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[0].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
417 
418  const T factor2 = ((channels[0].getAzimuth() - angle) / portion) * factor;
419  encoder.process(&factor2, vector_harmonics);
420  vector_harmonics[0] = factor2 * 0.5;
421  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[channels.size() - 1].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
422 
423  //post("portion : %f", (float)portion / HOA_2PI * 360.f);
424  //post("channel %i (%f) : %f", (int)channels[channels.size()-1].getIndex(),
425  // (float)(channels[channels.size()-1].getAzimuth() / HOA_2PI * 360.f),
426  // (float)(factor2 / factor) * 100.f);
427  //post("channel %i (%f) : %f", (int)channels[0].getIndex(),
428  // (float)(channels[0].getAzimuth() / HOA_2PI * 360.f),
429  // (float)(factor1 / factor) * 100.f);
430  }
431  else if(angle >= channels[channels.size() - 1].getAzimuth())
432  {
433  const T portion = (HOA_2PI - channels[channels.size()-1].getAzimuth()) + channels[0].getAzimuth();
434 
435  const T factor1 = (1. - ((angle - channels[channels.size()-1].getAzimuth()) / portion)) * factor;
436  encoder.setAzimuth(angle);
437  encoder.process(&factor1, vector_harmonics);
438  vector_harmonics[0] = factor1 * 0.5;
439  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[channels.size()-1].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
440 
441  const T factor2 = ((angle - channels[channels.size()-1].getAzimuth()) / portion) * factor;
442  encoder.process(&factor2, vector_harmonics);
443  vector_harmonics[0] = factor2 * 0.5;
444  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[0].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
445 
446  //post("portion : %f", (float)portion / HOA_2PI * 360.f);
447  //post("channel %i (%f) : %f", (int)channels[channels.size()-1].getIndex(),
448  // (float)(channels[channels.size()-1].getAzimuth() / HOA_2PI * 360.f),
449  // (float)(factor1 / factor) * 100.f);
450  //post("channel %i (%f) : %f", (int)channels[0].getIndex(),
451  // (float)(channels[0].getAzimuth() / HOA_2PI * 360.f),
452  // (float)(factor2 / factor) * 100.f);
453  }
454  else
455  {
456  for(size_t j = 1; j < channels.size(); j++)
457  {
458  if(angle < channels[j].getAzimuth() && angle >= channels[j-1].getAzimuth())
459  {
460  const T portion = (channels[j].getAzimuth() - channels[j-1].getAzimuth());
461 
462  const T factor1 = (1. - ((channels[j].getAzimuth() - angle) / portion)) * factor;
463  encoder.setAzimuth(angle);
464  encoder.process(&factor1, vector_harmonics);
465  vector_harmonics[0] = factor1 * 0.5;
466  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[j].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
467 
468  const T factor2 = ((channels[j].getAzimuth() - angle) / portion) * factor;
469  encoder.process(&factor2, vector_harmonics);
470  vector_harmonics[0] = factor2 * 0.5;
471  Signal<T>::add(Decoder<Hoa2d, T>::getNumberOfHarmonics(), vector_harmonics, m_matrix + channels[j-1].getIndex() * Decoder<Hoa2d, T>::getNumberOfHarmonics());
472 
473  //post("portion : %f", (float)portion / HOA_2PI * 360.f);
474  //post("channel %i (%f) : %f", (int)channels[j-1].getIndex(),
475  // (float)(channels[j-1].getAzimuth() / HOA_2PI * 360.f),
476  // (float)(factor2 / factor) * 100.f);
477  //post("channel %i (%f) : %f", (int)channels[j].getIndex(),
478  // (float)(channels[j].getAzimuth() / HOA_2PI * 360.f),
479  // (float)(factor1 / factor) * 100.f);
480 
481  break;
482  }
483  }
484  }
485  //post("");
486  }
487  channels.clear();
488  }
489  Signal<T>::free(vector_harmonics);
490  }

591  const size_t m = m_crop_size; // Impulses size
592  const size_t n = m_vector_size; // Vector size

597  }

611  input += m_vector_size;
612  }
613  processChannel(m_input, Hrir<Hoa2d, T>::getLeftMatrix(), m_left, outputs[0]);
614  processChannel(m_input, Hrir<Hoa2d, T>::getRightMatrix(), m_right, outputs[1]);
615  }
