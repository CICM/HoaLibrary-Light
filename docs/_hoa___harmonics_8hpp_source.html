113  static T getSemiNormalization(const size_t degree, const long order) hoa_noexcept {
114  if(D == Hoa2d) {
115  return 1.; }
116  const long double fac1 = hfactorial(long(degree) - long(std::abs(order)));
117  const long double fac2 = hfactorial(long(degree) + long(std::abs(order)));
118  return std::sqrt(static_cast<T>(bool(order == 0) ? 1. : 2.) * static_cast<T>(fac1 / fac2)); }
