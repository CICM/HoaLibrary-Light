89  inline void setFisheye(T fisheye) hoa_noexcept
90  {
91  const size_t nsources = getNumberOfSources();
92  const T factor = 1. - ((fisheye > T(1.)) ? T(1.) : ((fisheye < T(0.)) ? T(0.) : fisheye));
93  for(size_t i = 0; i < nsources; ++i)
94  {
95  const T azimuth = static_cast<T>(i) / static_cast<T>(nsources) * static_cast<T>(HOA_2PI);
96  if(azimuth < T(HOA_PI)) {
97  m_encoders[i].setAzimuth(azimuth * factor);
98  }
99  else {
100  m_encoders[i].setAzimuth(T(HOA_2PI) - ((T(HOA_2PI) - azimuth) * factor));
101  }
102  }

136  inline void process(const T* input, T* outputs) hoa_noexcept hoa_override
137  {
138  const size_t nencoders = m_encoders.size();
139  const size_t nharmos = ProcessorHarmonics<D, T>::getNumberOfHarmonics();
140  if(m_encoders[0].getMute())
141  {
142  memset(outputs, 0, nharmos * sizeof(T));
143  }
144  else
145  {
146  m_encoders[0].process(input, outputs);
147  }
148 
149  for(size_t i = 1; i < nencoders; i++)
150  {
151  m_encoders[i].process(++input, m_temp);
152  sigadd(nharmos, m_temp, outputs);
153  }
154  }

157  static inline void sigadd(const size_t size, const T* in, T* out) hoa_noexcept
158  {
159  for(size_t i = size>>3; i; --i, in += 8, out += 8)
160  {
161  out[0] += in[0]; out[1] += in[1]; out[2] += in[2]; out[3] += in[3];
162  out[4] += in[4]; out[5] += in[5]; out[6] += in[6]; out[7] += in[7];
163  }
164  for(size_t i = size&7; i; --i, in++, out++)
165  {
166  out[0] += in[0];
167  }
168  }
