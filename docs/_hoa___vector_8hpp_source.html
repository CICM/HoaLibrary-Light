123  inline void processVelocity(const T* inputs, T* outputs) hoa_noexcept
124  {
125  T veclocitySum = inputs[0];
126  for(size_t i = 1; i < ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves(); i++)
127  veclocitySum += inputs[i];
128 
129  const T velocityAbscissa = Signal<T>::dot(ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves(), inputs, m_channels_abscissa);
130  const T velocityOrdinate = Signal<T>::dot(ProcessorPlanewaves<Hoa2d, T>::getNumberOfPlanewaves(), inputs, m_channels_ordinate);
131  if(veclocitySum)
132  {
133  outputs[0] = velocityAbscissa / veclocitySum;
134  outputs[1] = velocityOrdinate / veclocitySum;
135  }
136  else
137  {
138  outputs[0] = outputs[1] = 0.;
139  }
140  }
