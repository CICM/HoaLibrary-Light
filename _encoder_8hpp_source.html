426  {
427  if(!m_muted)

434  (*outputs++) = (*input) * cos_x; // Hamonic [1,1]

439  tcos_x = cos_x;
440  (*outputs++) = (*input) * sin_x; // Hamonic [l,-l]

442  }
443  }

591  inline void process(const T* input, T* outputs) noexcept
592  {
593  if(!m_muted)
594  {
595  T cos_x = m_cosx;
596  T sin_x = m_sinx;
597  T tcos_x = cos_x;
598  const T gain1 = (m_gain * Processor<Hoa2d, T>::Harmonics::getDecompositionOrder());
599  const T factor1 = (cos(Math<T>::clip(m_factor, 0., HOA_PI)) + 1.) * 0.5 * ((gain1 - m_gain) + m_distance);
600 
601  (*outputs++) = (*input) * (gain1 + m_distance); // Hamonic [0,0]
602  (*outputs++) = (*input) * sin_x * factor1; // Hamonic [1,-1]
603  (*outputs++) = (*input) * cos_x * factor1; // Hamonic [1,1]
604  for(ulong i = 2; i <= Processor<Hoa2d, T>::Harmonics::getDecompositionOrder(); i++)
605  {
606  const T gain = (m_gain * (Processor<Hoa2d, T>::Harmonics::getDecompositionOrder() - i) + m_distance);
607  const T factor = (cos(Math<T>::clip(m_factor * i, 0., HOA_PI)) + 1.) * 0.5 ;
608 
609  cos_x = tcos_x * m_cosx - sin_x * m_sinx;
610  sin_x = tcos_x * m_sinx + sin_x * m_cosx;
611  tcos_x = cos_x;
612 
613  (*outputs++) = (*input) * sin_x * factor * gain; // Hamonic [i,-i]
614  (*outputs++) = (*input) * cos_x * factor * gain; // Hamonic [i,i]
615  }
616  }
617  else
618  {
619  for(ulong i = 0; i < Processor<Hoa2d, T>::Harmonics::getNumberOfHarmonics(); i++)
620  {
621  (*outputs++) = 0.;
622  }
623  }

956  void process(const T* input, T* outputs) noexcept
957  {
958  if(!m_muted)
959  {
960  const ulong order = Processor<Hoa3d, T>::Harmonics::getDecompositionOrder();
961  const T cos_theta = m_cos_theta;
962  const T sqr_theta = -m_sqrt_rmin;
963  const T cos_phi = (m_elevation >= -HOA_PI2 && m_elevation <= HOA_PI2) ? m_cos_phi : -m_cos_phi;
964  const T sin_phi = (m_elevation >= -HOA_PI2 && m_elevation <= HOA_PI2) ? m_sin_phi : -m_sin_phi;
965  const T* norm = m_normalization;
966 
967  // Elevation
968  T leg_l2 = 1.;
969  T leg_l1 = cos_theta;
970  T tleg_l;
971  T pleg_l = leg_l2;
972 
973  // Azimtuh
974  T cos_x = cos_phi;
975  T sin_x = sin_phi;
976  T tcos_x = cos_x;
977 
978  // For m[0] and l{0...N}
979  *(outputs) = (*input); // Hamonic [0, 0]
980  *(outputs+2) = (*input) * leg_l1; // Hamonic [1, 0]
981  ulong index = 6;
982  for(ulong i = 2; i <= order; i++, index += 2 * i)
983  {
984  tleg_l = (cos_theta * leg_l1 * (T)(2 * (i - 1) + 1) - (T)(i - 1) * leg_l2) / (T)(i);
985  leg_l2 = leg_l1;
986  leg_l1 = tleg_l;
987  *(outputs+index) = (*input) * leg_l1; // Hamonic [i, 0]
988  }
989 
990  // For m{1...N-1} and l{m...N}
991  index = 1;
992  for(ulong i = 1; i < order; i++, index = i * i)
993  {
994  ulong inc = 2;
995  leg_l2 = sqr_theta * pleg_l * (T)(2 * (i - 1) + 1);
996  leg_l1 = cos_theta * leg_l2 * (T)(2 * i + 1);
997  pleg_l = leg_l2;
998 
999  *(outputs+index) = (*input) * leg_l2 * sin_x * *(norm+index); // Hamonic [i,-i]
1000  index += 2*i;
1001  *(outputs+index) = (*input) * leg_l2 * cos_x * *(norm+index); // Hamonic [i, i]
1002  index += inc;
1003 
1004  *(outputs+index) = (*input) * leg_l1 * sin_x * *(norm+index); // Hamonic [i+1,-i]
1005  index += 2*i;
1006  *(outputs+index) = (*input) * leg_l1 * cos_x * *(norm+index); //Hamonic [i+1, i]
1007  inc += 2;
1008  index += inc;
1009 
1010  for(ulong j = i + 2; j <= order; j++)
1011  {
1012  tleg_l = (cos_theta * leg_l1 * (T)(2 * (j - 1) + 1) - (T)(j - 1 + i) * leg_l2) / (T)(j - i);
1013  leg_l2 = leg_l1;
1014  leg_l1 = tleg_l;
1015 
1016  *(outputs+index) = (*input) * leg_l1 * sin_x * *(norm+index); //Hamonic [j,-i]
1017  index += 2*i;
1018  *(outputs+index) = (*input) * leg_l1 * cos_x * *(norm+index); //Hamonic [j, i]
1019  inc += 2;
1020  index += inc;
1021  }
1022  cos_x = tcos_x * cos_phi - sin_x * sin_phi;
1023  sin_x = tcos_x * sin_phi + sin_x * cos_phi;
1024  tcos_x = cos_x;
1025  }
1026 
1027  // For m[N] and l[N]
1028  index = order * order;
1029  leg_l2 = sqr_theta * pleg_l * (T)(2 * (order - 1) + 1);
1030  *(outputs+index) = (*input) * leg_l2 * sin_x * *(norm+index);
1031  index += 2 * order;
1032  *(outputs+index) = (*input) * leg_l2 * cos_x * *(norm+index);
1033  }
1034  else
1035  {
1036  for(ulong i = 0; i < Processor<Hoa3d, T>::Harmonics::getNumberOfHarmonics(); i++)
1037  {
1038  (*outputs++) = 0.;
1039  }
1040  }
1041  }
