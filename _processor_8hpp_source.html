149  template <Dimension D, typename T> class Processor<D, T>::Planewaves : virtual public Processor<D, T>
150  {
151  private:
152  const ulong m_number_of_planewaves;
153  vector<Planewave<D, T> > m_planewaves;
154  T m_rotation_z;
155  T m_rotation_y;
156  T m_rotation_x;
157 
158  public:
159 
161 
164  Planewaves(const ulong numberOfPlanewaves) noexcept :
165  m_number_of_planewaves(numberOfPlanewaves),
166  m_rotation_z(0.),
167  m_rotation_y(0.),
168  m_rotation_x(0.)
169  {
170 #ifndef DOXYGEN_SHOULD_SKIP_THIS
171  if(D == Hoa2d)
172  {
173  for(ulong i = 0; i < m_number_of_planewaves; i++)
174  {
175  m_planewaves.push_back(Planewave<D, T>(i+1, (T)i / (m_number_of_planewaves) * HOA_2PI, 0.));
176  }
177  }
178  else
179  {
180  if(m_number_of_planewaves == 4)
181  {
182  const double oh = -(sqrt(2. / 3.) / sqrt(3. / 8.) - 1.);
183  const double hc = sqrt(1. - oh * oh);
184  const double el = asin(oh / sqrt(hc*hc + oh*oh));
185  m_planewaves.push_back(Planewave<D, T>(1ul, 0., HOA_PI2));
186  m_planewaves.push_back(Planewave<D, T>(2ul, 0., el));
187  m_planewaves.push_back(Planewave<D, T>(3ul, HOA_2PI / 3., el));
188  m_planewaves.push_back(Planewave<D, T>(4ul, 2. * HOA_2PI / 3., el));
189  }
190  else if(m_number_of_planewaves == 6)
191  {
192  m_planewaves.push_back(Planewave<D, T>(1ul, 0., HOA_PI2));
193  m_planewaves.push_back(Planewave<D, T>(2ul, 0., 0.));
194  m_planewaves.push_back(Planewave<D, T>(3ul, HOA_PI2, 0.));
195  m_planewaves.push_back(Planewave<D, T>(4ul, 2. * HOA_PI2, 0.));
196  m_planewaves.push_back(Planewave<D, T>(5ul, 3. * HOA_PI2, 0.));
197  m_planewaves.push_back(Planewave<D, T>(6ul, 0., -HOA_PI2));
198  }
199  else if(m_number_of_planewaves == 8)
200  {
201  m_planewaves.push_back(Planewave<D, T>(1ul, 1., 1., 1.));
202  m_planewaves.push_back(Planewave<D, T>(2ul, -1., 1., 1.));
203  m_planewaves.push_back(Planewave<D, T>(3ul, -1., -1., 1.));
204  m_planewaves.push_back(Planewave<D, T>(4ul, 1., -1., 1.));
205 
206  m_planewaves.push_back(Planewave<D, T>(5ul, 1., 1., -1.));
207  m_planewaves.push_back(Planewave<D, T>(6ul, -1., 1., -1.));
208  m_planewaves.push_back(Planewave<D, T>(7ul, -1., -1., -1.));
209  m_planewaves.push_back(Planewave<D, T>(8ul, 1., -1., -1.));
210  }
211  else if(m_number_of_planewaves == 12)
212  {
213  m_planewaves.push_back(Planewave<D, T>(1, 0., HOA_PI2));
214  for(ulong i = 1; i < 6; i++)
215  {
216  m_planewaves.push_back(Planewave<D, T>(i*2, T(i - 1.) / 5. * HOA_2PI, atan(0.5)));
217  m_planewaves.push_back(Planewave<D, T>(i*2+1, T(i - 1.) / 5. * HOA_2PI - HOA_PI / 5., -atan(0.5)));
218  }
219  m_planewaves.push_back(Planewave<D, T>(12, 0., -HOA_PI2));
220  }
221  else if(m_number_of_planewaves == 20)
222  {
223  const T phi = (sqrt(5.) - 1.) / 2.;
224  const T R = 1. / sqrt(3.);
225  const T a = R;
226  const T b = R / phi;
227  const T c = R * phi;
228  ulong index = 1;
229  for(long i = -1; i < 2; i += 2)
230  {
231  for(long j = -1; j < 2; j += 2)
232  {
233  m_planewaves.push_back(Planewave<D, T>(index++, 0., i * c * R, -j * b * R));
234  m_planewaves.push_back(Planewave<D, T>(index++, i * c * R, j * b * R, 0.));
235  m_planewaves.push_back(Planewave<D, T>(index++, i * b * R, 0., -j * c * R));
236  for(long k = -1; k < 2; k += 2)
237  {
238  m_planewaves.push_back(Planewave<D, T>(index++, i * a * R, j * a * R, -k * a * R));
239  }
240  }
241  }
242  }
243  else
244  {
245  if(m_number_of_planewaves % 2)
246  {
247  m_planewaves.push_back(Planewave<D, T>(1, 0., HOA_PI2));
248  }
249  const T phi = (sqrt(5.) - 1.) / 4.;
250  const ulong limit = (m_number_of_planewaves - (m_number_of_planewaves % 2)) / 2;
251  const T offset = 1. / T(limit) * HOA_PI;
252  for(ulong i = 0; i < m_number_of_planewaves - (m_number_of_planewaves % 2); i++)
253  {
254  if(i < limit)
255  {
256  m_planewaves.push_back(Planewave<D, T>(i+1, T(i) / T(limit) * HOA_2PI, phi));
257  }
258  else
259  {
260  m_planewaves.push_back(Planewave<D, T>(i+1, T(i - (limit)) / T(limit) * HOA_2PI + offset, phi * 2));
261  }
262  }
263  }
264  }
265 #endif
266  }
