840  void shiftAbscissa(double abscissa)
841  {
842  if(m_maximum_radius >= 0)
843  {
844  if(abscissa < 0.)
845  {
846  double refValue = -m_maximum_radius * 2.;
847  for (source_iterator it = m_sources.begin() ; it != m_sources.end() ; it ++)
848  {
849  double circleValue = -sqrt(m_maximum_radius * m_maximum_radius - it->second->getOrdinate() * it->second->getOrdinate());
850  if(circleValue - it->second->getAbscissa() > refValue)
851  refValue = circleValue - it->second->getAbscissa();
852  }
853  if(abscissa < refValue)
854  {
855  abscissa = refValue;
856  }
857  }
858  else if(abscissa >= 0.)
859  {
860  double refValue = m_maximum_radius * 2.;
861  for (source_iterator it = m_sources.begin() ; it != m_sources.end() ; it ++)
862  {
863  double circleValue = sqrt(m_maximum_radius * m_maximum_radius - it->second->getOrdinate() * it->second->getOrdinate());
864  if(circleValue - it->second->getAbscissa() < refValue)
865  refValue = circleValue - it->second->getAbscissa();
866  }
867  }
868  }
869  for (source_iterator it = m_sources.begin() ; it != m_sources.end() ; it ++)
