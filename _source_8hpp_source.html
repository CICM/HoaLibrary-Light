841  void shiftAbscissa(double abscissa)
842  {
843  if(m_maximum_radius >= 0)
844  {
845  if(abscissa < 0.)
846  {
847  double refValue = -m_maximum_radius * 2.;
848  for (source_iterator it = m_sources.begin() ; it != m_sources.end() ; it ++)
849  {
850  double circleValue = -sqrt(m_maximum_radius * m_maximum_radius - it->second->getOrdinate() * it->second->getOrdinate());
851  if(circleValue - it->second->getAbscissa() > refValue)
852  refValue = circleValue - it->second->getAbscissa();
853  }
854  if(abscissa < refValue)
855  {
856  abscissa = refValue;
857  }
858  }
859  else if(abscissa >= 0.)
860  {
861  double refValue = m_maximum_radius * 2.;
862  for (source_iterator it = m_sources.begin() ; it != m_sources.end() ; it ++)
863  {
864  double circleValue = sqrt(m_maximum_radius * m_maximum_radius - it->second->getOrdinate() * it->second->getOrdinate());
865  if(circleValue - it->second->getAbscissa() < refValue)
866  refValue = circleValue - it->second->getAbscissa();
867  }
868  }
869  }
870  for (source_iterator it = m_sources.begin() ; it != m_sources.end() ; it ++)
