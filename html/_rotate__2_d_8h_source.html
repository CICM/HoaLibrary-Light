71  inline void process(const float* inputs, float* outputs) const noexcept
72  {
73  float cos_x = m_cosx;
74  float sin_x = m_sinx;
75  float tcos_x = cos_x;
76  float sig;
77  outputs[0] = inputs[0];
78  for(unsigned long i = 2; i < m_number_of_harmonics; i += 2)
79  {
80  sig = inputs[i-1];
81  outputs[i-1] = sin_x * inputs[i] + cos_x * sig;
82  outputs[i] = cos_x * inputs[i] - sin_x * sig;
83  cos_x = tcos_x * m_cosx - sin_x * m_sinx; // cos(x + b) = cos(x) * cos(b) - sin(x) * sin(b)
84  sin_x = tcos_x * m_sinx + sin_x * m_cosx; // sin(x + b) = cos(x) * sin(b) + sin(x) * cos(b)
85  tcos_x = cos_x;
86  }
