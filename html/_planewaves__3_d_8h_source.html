260  inline void sphere_discretize(unsigned int numberOfPoints, double* azimuths, double* elevations)
261  {
262  if(numberOfPoints == 4) // Tethrahedron
263  {
264  double oh = sqrt(2. / 3.) / sqrt(3. / 8.) - 1;
265  double hc = sqrt(1 - oh * oh);
266  azimuths[0] = 0.;
267  elevations[0] = HOA_PI2;
268  azimuths[1] = 0.;
269  azimuths[2] = HOA_2PI / 3.;
270  azimuths[3] = 2. * HOA_2PI / 3.;
271  elevations[1] = elevations[2] = elevations[3] = elevation(0, hc, -oh);
272  }
273  else if(numberOfPoints == 5) // Not regular : Octahedron with only 3 points at 0 azymuth
274  {
275  azimuths[0] = 0.;
276  elevations[0] = HOA_PI2;
277  azimuths[1] = 0.;
278  azimuths[2] = HOA_2PI / 3.;
279  azimuths[3] = 2. * HOA_2PI / 3.;
280  elevations[1] = elevations[2] = elevations[3] = 0;
281  azimuths[4] = 0.;
282  elevations[4] = -HOA_PI2;
283  }
284  else if(numberOfPoints == 6) // Octahedron
285  {
286  azimuths[0] = 0.;
287  elevations[0] = HOA_PI2;
288  azimuths[1] = 0.;
289  azimuths[2] = HOA_PI2;
290  azimuths[3] = 2. * HOA_PI2;
291  azimuths[4] = 3. * HOA_PI2;
292  elevations[1] = elevations[2] = elevations[3] = elevations[4] = 0;
293  azimuths[5] = 0.;
294  elevations[5] = -HOA_PI2;
295  }
296  else if(numberOfPoints == 7) // Not regular : Octahedron with only 5 points at 0 azymuth
297  {
298  azimuths[0] = 0.;
299  elevations[0] = HOA_PI2;
300  azimuths[1] = 0.;
301  azimuths[2] = HOA_2PI / 5.;
302  azimuths[3] = 2. * HOA_2PI / 5.;
303  azimuths[4] = 3. * HOA_2PI / 5.;
304  azimuths[5] = 4. * HOA_2PI / 5.;
305  elevations[1] = elevations[2] = elevations[3] = elevations[4] = elevations[5] = 0;
306  azimuths[6] = 0.;
307  elevations[6] = -HOA_PI2;
308  }
309  else if(numberOfPoints == 8) // Hexahedron or cube
310  {
311  double elev = asin(1/sqrt(3.)); // or elevation(-1, 1, 1);
312  azimuths[0] = azimuths[4] = HOA_PI4;
313  azimuths[1] = azimuths[5] = HOA_PI4 + HOA_PI2;
314  azimuths[2] = azimuths[6] = HOA_PI4 + HOA_PI;
315  azimuths[3] = azimuths[7] = HOA_PI4 + HOA_PI + HOA_PI2;
316  elevations[0] = elevations[1] = elevations[2] = elevations[3] = elev;
317  elevations[4] = elevations[5] = elevations[6] = elevations[7] = -elev;
318  }
319  else if(numberOfPoints == 9) // Not regular : Hexahedron or cube with 1 at π/2 elevation
320  {
321  azimuths[0] = azimuths[4] = HOA_PI4;
322  azimuths[1] = azimuths[5] = HOA_PI4 + HOA_PI2;
323  azimuths[2] = azimuths[6] = HOA_PI4 + HOA_PI;
324  azimuths[3] = azimuths[7] = HOA_PI4 + HOA_PI + HOA_PI2;
325  elevations[0] = elevations[1] = elevations[2] = elevations[3] = elevation(-1, 1, 1);
326  elevations[4] = elevations[5] = elevations[6] = elevations[7] = -elevations[3];
327  elevations[8] = HOA_PI2;
328  azimuths[8] = 0;
329  }
330  else if(numberOfPoints == 10) // Not regular : Hexahedron or cube with 1 at π/2 elevation and 1 at -π/2 elevation
331  {
332  azimuths[0] = azimuths[4] = HOA_PI4;
333  azimuths[1] = azimuths[5] = HOA_PI4 + HOA_PI2;
334  azimuths[2] = azimuths[6] = HOA_PI4 + HOA_PI;
335  azimuths[3] = azimuths[7] = HOA_PI4 + HOA_PI + HOA_PI2;
336  elevations[0] = elevations[1] = elevations[2] = elevations[3] = elevation(-1, 1, 1);
337  elevations[4] = elevations[5] = elevations[6] = elevations[7] = -elevations[3];
338  elevations[8] = HOA_PI2;
339  elevations[9] = -HOA_PI2;
340  azimuths[9] = azimuths[8] = 0;
341  }
342  else if(numberOfPoints == 11) // Not regular : Hexahedron or cube with 1 at π/2 elevation and 1 at -π/2 elevation
343  {
344  azimuths[0] = azimuths[4] = HOA_PI4;
345  azimuths[1] = azimuths[5] = HOA_PI4 + HOA_PI2;
346  azimuths[2] = azimuths[6] = HOA_PI4 + HOA_PI;
347  azimuths[3] = azimuths[7] = HOA_PI4 + HOA_PI + HOA_PI2;
348  elevations[0] = elevations[1] = elevations[2] = elevations[3] = elevation(-1, 1, 1);
349  elevations[4] = elevations[5] = elevations[6] = elevations[7] = -elevations[3];
350  elevations[8] = HOA_PI2;
351  elevations[9] = elevations[9] = -HOA_PI2;
352  azimuths[9] = azimuths[8] = 0;
353  }
354  else if(numberOfPoints == 12) // Icosahedron
355  {
356  azimuths[0] = 0;
357  elevations[0] = HOA_PI2;
358  azimuths[11] = 0;
359  elevations[11] = -HOA_PI2;
360  for(int i = 1; i < 6; i++)
361  {
362  azimuths[i] = (i - 1.) / 5. * HOA_2PI;
363  azimuths[i+5] = azimuths[i] - HOA_PI / 5.;
364  elevations[i] = atan(0.5);
365  elevations[i+5] = -elevations[i];
366  }
367  }
368  else if(numberOfPoints > 12 && numberOfPoints < 20)
369  {
370  double elev[20];
371  double azim[20];
372  sphere_discretize(20, azim, elev);
373  for(int i = 0; i < numberOfPoints; i++)
374  {
375  azimuths[i] = azim[i];
376  elevations[i] = elev[i];
377  }
378  }
379  else if(numberOfPoints == 20) // dodecahedron
380  {
381  double phi = (sqrt(5.) - 1.) / 2.; // The golden ratio
382  double R = 1. / sqrt(3.);
383 
384  double a = R;
385  double b = R / phi;
386  double c = R * phi;
387 
388  for(int i = -1, index = 0; i < 2; i += 2)
389  {
390  for(int j = -1; j < 2; j += 2)
391  {
392  azimuths[index] = azimuth(0., i * c * R, j * b * R);
393  elevations[index] = elevation(0., i * c * R, j * b * R) * -1;
394  index++;
395 
396  azimuths[index] = azimuth(i * c * R, j * b * R, 0.);
397  elevations[index] = elevation(i * c * R, j * b * R, 0.) * -1;
398  index++;
399 
400  azimuths[index] = azimuth(i * b * R, 0., j * c * R);
401  elevations[index] = elevation(i * b * R, 0., j * c * R) * -1;
402  index++;
403 
404  for (int k = -1; k < 2; k += 2)
405  {
406  azimuths[index] = azimuth(i * a * R, j * a * R, k * a * R);
407  elevations[index] = elevation(i * a * R, j * a * R, k * a * R) * -1;
408  index++;
409  }
410  }
411  }
412  }
413  else // Two circles
414  {
415  unsigned int i = 0;
416  int up = ceil(numberOfPoints * 0.5);
417  int down = numberOfPoints - up;
418  for(i = 0; i < up; i++)
419  {
420  azimuths[i] = (double)i / (double)up * HOA_2PI;
421  elevations[i] = HOA_PI4;
422  }
423  for(; i < numberOfPoints; i++)
424  {
425  azimuths[i] = (double)(i - up) / (double)down * HOA_2PI;
426  elevations[i] = -HOA_PI4;
427  }
428  }
429  }
