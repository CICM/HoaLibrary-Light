161  void setCoefficients(float* coeffs1, float* coeffs2, float factor, float* coeffs3, float factor2)
162  {
163  m_coeff_a0 = coeffs1[0] * factor + coeffs2[0] * (1.f - factor);
164  m_coeff_a1 = coeffs1[1] * factor + coeffs2[1] * (1.f - factor);

166  m_coeff_b1 = coeffs1[3] * factor + coeffs2[3] * (1.f - factor);
167  m_coeff_b2 = coeffs1[4] * factor + coeffs2[4] * (1.f - factor);
168  m_coeff_a0 = coeffs3[0] * factor2 + m_coeff_a0 * (1.f - factor2);
169  m_coeff_a1 = coeffs3[1] * factor2 + m_coeff_a1 * (1.f - factor2);
170  m_coeff_a2 = coeffs3[2] * factor2 + m_coeff_a2 * (1.f - factor2);
171  m_coeff_b1 = coeffs3[3] * factor2 + m_coeff_b1 * (1.f - factor2);
172  m_coeff_b2 = coeffs3[4] * factor2 + m_coeff_b2 * (1.f - factor2);
173  m_coeff_a0 *= m_sample_rate / 44100.f;
174  m_coeff_a1 *= m_sample_rate / 44100.f;
175  m_coeff_a2 *= m_sample_rate / 44100.f;

180  inline float process(float input)
181  {
182  float output = input * m_coeff_a0 + m_delay_one;

184  m_delay_two = input * m_coeff_a2 - m_coeff_b2 * output;
