126  inline void process(const Type input, Type* outputs) const noexcept
127  {
128  if(!m_muted)
129  {
130  Type cos_x = m_cosx;
131  Type sin_x = m_sinx;
132  Type tcos_x = cos_x;
133  outputs[0] = input * (m_gain * m_order_of_decomposition + 1.f);
134  for(unsigned long i = 2, j = 1; i < m_number_of_harmonics; i += 2, j++)
135  {
136  const Type factor = (std::cos(min(m_factor * j, (Type)HOA_PI)) + 1.f) * 0.5f * (m_gain * (m_order_of_decomposition - j) + 1.f);
137  outputs[i-1] = input * sin_x * factor;
138  outputs[i] = input * cos_x * factor;
139  cos_x = tcos_x * m_cosx - sin_x * m_sinx; // cos(x + b) = cos(x) * cos(b) - sin(x) * sin(b)
140  sin_x = tcos_x * m_sinx + sin_x * m_cosx; // sin(x + b) = cos(x) * sin(b) + sin(x) * cos(b)
141  tcos_x = cos_x;
142  }
143  }
144  else
145  {
146  for(unsigned long i = 0; i < m_number_of_harmonics; i++)
147  {
148  outputs[i] = 0.f;
149  }
150  }
151  }

158  inline void processAdd(const Type input, Type* outputs) noexcept
159  {
160  if(!m_muted)
161  {
162  Type cos_x = m_cosx;
163  Type sin_x = m_sinx;

165  outputs[0] = input * (m_gain * m_order_of_decomposition + 1.f);
166  for(unsigned long i = 2, j = 1; i < m_number_of_harmonics; i += 2, j++)
167  {
168  const Type factor = (std::cos(min(m_factor * j, (Type)HOA_PI)) + 1.f) * 0.5f * (m_gain * (m_order_of_decomposition - j) + 1.f);
169  outputs[i-1] += input * sin_x * factor;
170  outputs[i] += input * cos_x * factor;
171  cos_x = tcos_x * m_cosx - sin_x * m_sinx; // cos(x + b) = cos(x) * cos(b) - sin(x) * sin(b)
172  sin_x = tcos_x * m_sinx + sin_x * m_cosx; // sin(x + b) = cos(x) * sin(b) + sin(x) * cos(b)
173  tcos_x = cos_x;
