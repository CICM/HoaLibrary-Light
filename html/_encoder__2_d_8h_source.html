68  inline void process(const Type input, Type* outputs) const noexcept
69  {
70  float cos_x = m_cosx, tcos_x = m_cosx, sin_x = m_sinx;
71  outputs[0] = input; // Hamonic [0,0]
72  outputs[1] = input * cos_x; // Hamonic [1,-1]
73  outputs[2] = input * sin_x; // Hamonic [1,1]
74  for(unsigned long i = 3; i < m_number_of_harmonics; i += 2)
75  {
76  cos_x = tcos_x * m_cosx - sin_x * m_sinx; // cos(x + b) = cos(x) * cos(b) - sin(x) * sin(b)
77  sin_x = tcos_x * m_sinx + sin_x * m_cosx; // sin(x + b) = cos(x) * sin(b) + sin(x) * cos(b)
78  tcos_x = cos_x;
79  outputs[i] = input * sin_x; // Hamonic [l,-l]
80  outputs[i+1]= input * cos_x; // Hamonic [l,l]
81  }
82  }
