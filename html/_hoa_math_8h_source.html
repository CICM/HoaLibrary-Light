116  inline double associated_legendre(int l, int m, const double x)
117  {
118  l = abs(l);
119  m = abs(m);
120 
121  if(l == m)
122  {
123  return pow((double)-1.0, (double)m) * pow((double)(1. - x * x), (double)(0.5 * m)) * double_factorial(2. * m - 1);
124  }
125  // Calculus = pow((double)-1.0, (double)m) * pow((double)(1. - x * x), (double)(0.5 * m))
126  // Normalization = val / double_factorial(2. * m - 1)
127  else if(l == m + 1)
128  {
129  return x * associated_legendre(m, m, x) * (2 * m + 1);
130  }
131  // Calculus = x * associated_legendre(m, m, x)
132  // Normalization = val / double_factorial(2. * m) * (2 * m + 1);
133  else
134  {
135  return ((double)(2 * l - 1) * x * associated_legendre(l - 1, m, x) - (double)(l + m - 1.) * associated_legendre(l - 2, m, x)) / (double)(l - m);
136  // (l == m + 2) -> ((2 * (m + 2) - 1) * x * associated_legendre(m + 1, m, x) - (2 * m + 1) * associated_legendre(m, m, x)) /
137  }
138  // Real Calculus
139  // A = (2 * l - 1) * x * associated_legendre(l - 1, m, x) [ (2 * (l - 1) - 1) * (2 * (l - 1) + 1) ]
140  // B = (l + m - 1.) * associated_legendre(l - 2, m, x) [ (2 * (l - 1) - 1) * (2 * (l - 1) - 1) * (l + m - 1.) ]
141 
142  // Calculus = associated_legendre(l - 1, m, x) - associated_legendre(l - 2, m, x)

378  inline double spherical_azimuth_interpolation(const double azimuth1, const double elevation1, const double azimuth2, const double elevation2, double mu)
379  {
380  double distance;
381  double angle1 = wrap_twopi(azimuth1);
382  double angle2 = wrap_twopi(azimuth2);
383  if(angle1 > angle2)
384  distance = (angle1 - angle2);
385  else
386  distance = (angle2 - angle1);
387  if(azimuth1 == azimuth2)
388  return azimuth1;
