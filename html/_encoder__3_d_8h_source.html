116  void process(const float input, float* outputs) const noexcept
117  {
118  float leg1, leg2;
119  float hzero1 = m_cos_theta, hzero2 = 1.f;
120  float cos_xphi = m_cos_phi, tcos_xphi = m_cos_phi, sin_xphi = m_sin_phi;
121  outputs[0] = hzero2; // Hamonic [0,0] = 1
122  outputs[1] = -m_sqrt_rmin; // Hamonic [1,-1] = -√(1-x²) * sin(ϕ)
123  outputs[2] = hzero1; // Hamonic [1,0] = x
124  outputs[3] = -m_sqrt_rmin; // Hamonic [1,1] = -√(1-x²) * cos(ϕ)
125 
126  for(unsigned long i = 4, l = 2, h = 2 * l + 1; l <= m_order_of_decomposition; l++, h = 2 * l + 1)
127  {
128  leg1 = -m_sqrt_rmin * h * outputs[i-1];
129  leg2 = m_cos_theta * h * outputs[i-1];
130  outputs[i++]= leg1; // Hamonic [l,-l] = P(l+1,l+1)
131  outputs[i++]= leg2; // Hamonic [l,-l+1] = P(l,l+1)
132 
133  for(unsigned long m = l - 2, index = i - 3; m; m--, index--)
134  {
135  float leg = ((l - m) * m_cos_theta * outputs[i-1] - (l + m) * outputs[i-2l-1]) / m_sqrt_rmin;
136  outputs[i] = leg; // Hamonic [l,-m] = P(l,m+1)
137  outputs[i+m*2] = leg; // Hamonic [l,m] = P(l,m+1)
138  i++;
139  }
140  float leg = (m_cos_theta * h * hzero1 - l * hzero2) / (l + 1.f);
141  hzero2 = hzero1;
142  hzero1 = leg;
143  outputs[i+=l-1] = leg; // Hamonic [l,0] = P(l+1, 0)
144  outputs[i++] = leg2; // Hamonic [l,l-1] = P(l,l+1)
145  outputs[i++] = leg1; // Hamonic [l,l] = P(l+1,l+1)
146  }
147 
148  // Multiplication of the factors by the input
149  cblas_sscal(m_number_of_harmonics, input, outputs, 1);
150  }
