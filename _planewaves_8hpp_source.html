392  inline T getAbscissa(const T x_axe, const T y_axe, const T z_axe) const noexcept
393  {
394  T x = cos(m_azimuth + HOA_PI2) * cos(m_elevation);
395  T y = sin(m_azimuth + HOA_PI2) * cos(m_elevation);
396  T z = sin(m_elevation);
397 
398  const T cosAngle = cos(x_axe);
399  const T sinAngle = sin(x_axe);
400  T ry = y * cosAngle - z * sinAngle;
401  z = y * sinAngle + z * cosAngle;
402  y = ry;
403 
404  x = x * cos(z_axe) - y * sin(z_axe);
405 
406  return x * cos(y_axe) - z * sin(y_axe);
407  }
