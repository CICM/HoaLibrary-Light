192  Scope(ulong order, ulong numberOfRow, ulong numberOfColumn) noexcept :
193  Encoder<Hoa3d, T>::Basic(order),
194  Processor<Hoa3d, T>::Planewaves(numberOfRow * numberOfColumn),
195  m_number_of_rows(numberOfRow),
196  m_number_of_columns(numberOfColumn)
197  {
198  for(ulong i = 0; i < m_number_of_rows; i++)
199  {
200  const T elevation = (T)i * HOA_PI / (T)(m_number_of_rows - 1) - HOA_PI2;
201  for(ulong j = 0; j < m_number_of_columns; j++)
202  {
203  Processor<Hoa3d, T>::Planewaves::setPlanewaveAzimuth(i * m_number_of_columns + j, (T)j * HOA_2PI / (T)m_number_of_columns);
204  Processor<Hoa3d, T>::Planewaves::setPlanewaveElevation(i * m_number_of_columns + j, elevation);
205  }
206  }
207 
208  m_matrix = new T[Processor<Hoa3d, T>::Planewaves::getNumberOfPlanewaves() * Encoder<Hoa3d, T>::getNumberOfHarmonics()];
209  m_vector = new T[Processor<Hoa3d, T>::Planewaves::getNumberOfPlanewaves()];
210 
211  const T factor = 12.5 / (T)(Encoder<Hoa3d, T>::getNumberOfHarmonics());
212  for(ulong i = 0; i < Processor<Hoa3d, T>::Planewaves::getNumberOfPlanewaves(); i++)
213  {
214  Encoder<Hoa3d, T>::Basic::setAzimuth(Processor<Hoa3d, T>::Planewaves::getPlanewaveAzimuth(i));
215  Encoder<Hoa3d, T>::Basic::setElevation(Processor<Hoa3d, T>::Planewaves::getPlanewaveElevation(i));
216  Encoder<Hoa3d, T>::Basic::process(&factor, m_matrix + i * Encoder<Hoa3d, T>::getNumberOfHarmonics());
217  for(ulong j = 0; j < Encoder<Hoa3d, T>::getNumberOfHarmonics(); j++)
218  {
219  const ulong l = Encoder<Hoa3d, T>::getHarmonicDegree(j);
220  m_matrix[i * Encoder<Hoa3d, T>::getNumberOfHarmonics() + j] *= (2. * l + 1.) / (4. * HOA_PI);
221  }
222  }
223  for(ulong i = 0; i < Processor<Hoa3d, T>::Planewaves::getNumberOfPlanewaves(); i++)
224  {
225  m_vector[i] = 0.;
226  }
227  m_maximum = 0;
